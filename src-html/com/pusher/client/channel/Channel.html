001    package com.pusher.client.channel;
002    
003    /**
004     * An object that represents a Pusher channel. An implementation of this interface is returned when you call
005     * {@link com.pusher.client.Pusher#subscribe(String)} or {@link com.pusher.client.Pusher#subscribe(String, ChannelEventListener, String...)}.
006     * 
007     */
008    public interface Channel {
009    
010        /**
011         * Gets the name of the Pusher channel that this object represents.
012         * @return The name of the channel.
013         */
014        String getName();
015        
016        /**
017         * Binds a {@link SubscriptionEventListener} to an event. The {@link SubscriptionEventListener} will be notified whenever the
018         * specified event is received on this channel.
019         * 
020         * @param eventName The name of the event to listen to.
021         * @param listener A listener to receive notifications when the event is received.
022         * @throws IllegalArgumentException If either of the following are true:
023         * <ul>
024         *  <li>The name of the event is null.</li>
025         *  <li>The {@link SubscriptionEventListener} is null.</li>
026         * </ul>
027         * @throws IllegalStateException If the channel has been unsubscribed by calling {@link com.pusher.client.Pusher#unsubscribe(String)}. 
028         * This puts the {@linkplain Channel} in a terminal state from which it can no longer be used. To resubscribe, call 
029         *  {@link com.pusher.client.Pusher#subscribe(String)} or {@link com.pusher.client.Pusher#subscribe(String, ChannelEventListener, String...)}
030         *  again to receive a fresh {@linkplain Channel} instance.
031         */
032        void bind(String eventName, SubscriptionEventListener listener);
033        
034        /**
035         * <p>Unbinds a previously bound {@link SubscriptionEventListener} from an event. The {@link SubscriptionEventListener} will no
036         * longer be notified whenever the specified event is received on this channel.</p> 
037         * 
038         * <p>Calling this method does not unsubscribe from the channel even if there are no more {@link SubscriptionEventListener}s 
039         * bound to it. If you want to unsubscribe from the channel completely, call {@link com.pusher.client.Pusher#unsubscribe(String)}.
040         * It is not necessary to unbind your {@link SubscriptionEventListener}s first.</p>
041         * 
042         * @param eventName The name of the event to stop listening to.
043         * @param listener The listener to unbind from the event.
044         * @throws IllegalArgumentException If either of the following are true:
045         * <ul>
046         *  <li>The name of the event is null.</li>
047         *  <li>The {@link SubscriptionEventListener} is null.</li>
048         * </ul>
049         * @throws IllegalStateException If the channel has been unsubscribed by calling {@link com.pusher.client.Pusher#unsubscribe(String)}. 
050         * This puts the {@linkplain Channel} in a terminal state from which it can no longer be used. To resubscribe, call 
051         *  {@link com.pusher.client.Pusher#subscribe(String)} or {@link com.pusher.client.Pusher#subscribe(String, ChannelEventListener, String...)}
052         *  again to receive a fresh {@linkplain Channel} instance.
053         */
054        void unbind(String eventName, SubscriptionEventListener listener);
055    }
