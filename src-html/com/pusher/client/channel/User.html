001    package com.pusher.client.channel;
002    
003    import com.google.gson.Gson;
004    
005    /**
006     * Represents a user that is subscribed to a
007     * {@link com.pusher.client.channel.PresenceChannel PresenceChannel}.
008     */
009    public class User {
010    
011        private final String id;
012        private final String jsonData;
013    
014        /**
015         * Create a new user. Users should not be created within an application. Users are created within the library and represent subscriptions to presence channels.
016         * @param id
017         * @param jsonData
018         */
019        public User(String id, String jsonData) {
020            this.id = id;
021            this.jsonData = jsonData;
022        }
023    
024        /**
025         * A unique identifier for the user within a Pusher application.
026         * @return The unique id.
027         */
028        public String getId() {
029            return id;
030        }
031    
032        /**
033         * Custom additional information about a user as a String encoding a JSON hash
034         * @return
035         */
036        public String getInfo() {
037            return jsonData;
038        }
039    
040        /**
041         * Custom additional information about a user decoded as a new instance of the provided POJO bean type
042         *
043         * e.g. if <pre>getInfo()</pre> returns <pre>{"name":"Mr User","number":9}</pre>
044         * <pre>
045           public class UserInfo {
046               private String name;
047               private Integer number;
048    
049               public String getName() {
050                   return name;
051               }
052               public void setName(String name) {
053                   this.name = name;
054               }
055               public Integer getNumber() {
056                   return number;
057               }
058               public void setNumber(Integer number) {
059                   this.number = number;
060               }
061           }
062    
063           UserInfo info = user.getInfo(UserInfo.class);
064           info.getName() // returns "Mr User"
065           info.getNumber() // return 9
066           </pre>
067         *
068         * @param clazz the class into which the user info JSON representation should be parsed.
069         * @return An instance of clazz, populated with the user info
070         */
071        public <V> V getInfo(Class<V> clazz) {
072            return new Gson().fromJson(jsonData, clazz);
073        }
074    
075        @Override
076        public String toString() {
077            return String.format("[User id=%s, data=%s]", id, jsonData);
078        }
079    
080        @Override
081        public int hashCode() {
082            return id.hashCode() + ((jsonData != null) ? jsonData.hashCode() : 0);
083        }
084    
085        @Override
086        public boolean equals(Object other) {
087    
088            if (other instanceof User) {
089                User otherUser = (User) other;
090                return this.getId().equals(otherUser.getId())
091                        && this.getInfo().equals(otherUser.getInfo());
092            }
093    
094            return false;
095        }
096    }
