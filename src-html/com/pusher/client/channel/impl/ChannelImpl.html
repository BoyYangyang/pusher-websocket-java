001    package com.pusher.client.channel.impl;
002    
003    import java.util.HashMap;
004    import java.util.HashSet;
005    import java.util.LinkedHashMap;
006    import java.util.Map;
007    import java.util.Set;
008    
009    import com.google.gson.Gson;
010    import com.pusher.client.channel.ChannelEventListener;
011    import com.pusher.client.channel.ChannelState;
012    import com.pusher.client.channel.SubscriptionEventListener;
013    import com.pusher.client.util.Factory;
014    
015    public class ChannelImpl implements InternalChannel {
016    
017        private static final String INTERNAL_EVENT_PREFIX = "pusher_internal:";
018        protected static final String SUBSCRIPTION_SUCCESS_EVENT = "pusher_internal:subscription_succeeded";
019        protected final String name;
020        protected final Map<String, Set<SubscriptionEventListener>> eventNameToListenerMap = new HashMap<String, Set<SubscriptionEventListener>>();
021        protected ChannelState state = ChannelState.INITIAL;
022            private ChannelEventListener eventListener;
023    
024        public ChannelImpl(String channelName) {
025            
026            if(channelName == null) {
027                throw new IllegalArgumentException("Cannot subscribe to a channel with a null name");
028            }
029            
030            for(String disallowedPattern : getDisallowedNameExpressions()) {
031                if(channelName.matches(disallowedPattern)) {
032                    throw new IllegalArgumentException("Channel name " + channelName + " is invalid. Private channel names must start with \"private-\" and presence channel names must start with \"presence-\"");
033                }
034            }
035            
036            this.name = channelName;
037        }
038    
039        /* Channel implementation */
040        
041        @Override
042        public String getName() {
043            return name;
044        }
045        
046        @Override
047        public void bind(String eventName, SubscriptionEventListener listener) {
048            
049            validateArguments(eventName, listener);
050            
051            Set<SubscriptionEventListener> listeners = eventNameToListenerMap.get(eventName);
052            if(listeners == null) {
053                listeners = new HashSet<SubscriptionEventListener>();
054                eventNameToListenerMap.put(eventName, listeners);
055            }
056            
057            listeners.add(listener);
058        }
059    
060        @Override
061        public void unbind(String eventName, SubscriptionEventListener listener) {
062            
063            validateArguments(eventName, listener);
064            
065            Set<SubscriptionEventListener> listeners = eventNameToListenerMap.get(eventName);
066            if(listeners != null) {
067                listeners.remove(listener);
068                if(listeners.isEmpty()) {
069                    eventNameToListenerMap.remove(eventName);
070                }
071            }
072        }
073        
074        /* InternalChannel implementation */
075       
076        @Override
077        public void onMessage(final String event, String message) {
078    
079            if(event.equals(SUBSCRIPTION_SUCCESS_EVENT)) {
080                updateState(ChannelState.SUBSCRIBED);
081            } else {
082                Set<SubscriptionEventListener> listeners = eventNameToListenerMap.get(event);
083                if(listeners != null) {
084                    for(final SubscriptionEventListener listener : listeners) {
085                            
086                        final String data = extractDataFrom(message);
087                            
088                        Factory.getEventQueue().execute(new Runnable() {
089                            public void run() {
090                                listener.onEvent(name, event, data);
091                            }
092                        });
093                    }
094                }
095            }
096        }
097        
098        @Override
099        public String toSubscribeMessage() {
100            
101            Map<Object, Object> jsonObject = new LinkedHashMap<Object, Object>();
102            jsonObject.put("event", "pusher:subscribe");
103            
104            Map<Object, Object> dataMap = new LinkedHashMap<Object, Object>();
105            dataMap.put("channel", name);
106            
107            jsonObject.put("data", dataMap);
108            
109            return new Gson().toJson(jsonObject);
110        }
111        
112        @Override
113        public String toUnsubscribeMessage() {
114            Map<Object, Object> jsonObject = new LinkedHashMap<Object, Object>();
115            jsonObject.put("event", "pusher:unsubscribe");
116            
117            Map<Object, Object> dataMap = new LinkedHashMap<Object, Object>();
118            dataMap.put("channel", name);
119            
120            jsonObject.put("data", dataMap);
121            
122            return new Gson().toJson(jsonObject);
123        }
124    
125        @Override
126        public void updateState(ChannelState state) {
127            
128            this.state = state;
129            
130            if(state == ChannelState.SUBSCRIBED && eventListener != null) {
131                Factory.getEventQueue().execute(new Runnable() {
132                            public void run() {
133                                eventListener.onSubscriptionSucceeded(ChannelImpl.this.getName());
134                            }
135                        });
136                    }
137        }
138    
139        /* Comparable implementation */
140    
141            @Override
142            public void setEventListener(ChannelEventListener listener) {
143                    this.eventListener = listener;
144            }
145    
146            @Override
147            public ChannelEventListener getEventListener() {
148                    return eventListener;
149            }
150        
151        @Override
152        public int compareTo(InternalChannel other) {
153            return this.getName().compareTo(other.getName());
154        }
155        
156        /* implementation detail */
157        
158        @Override
159        public String toString() {
160            return String.format("[Public Channel: name=%s]", name);
161        }
162        
163        @SuppressWarnings("unchecked")
164        private String extractDataFrom(String message) {
165            Gson gson = new Gson();
166            Map<Object, Object> jsonObject = gson.fromJson(message, Map.class);
167            return gson.toJson(jsonObject.get("data"));
168        }
169    
170        protected String[] getDisallowedNameExpressions() {
171            return new String[] {
172                    "^private-.*",
173                    "^presence-.*"
174            };
175        }
176        
177        private void validateArguments(String eventName, SubscriptionEventListener listener) {
178            
179            if(eventName == null) {
180                throw new IllegalArgumentException("Cannot bind or unbind to channel " + name + " with a null event name");
181            }
182            
183            if(listener == null) {
184                throw new IllegalArgumentException("Cannot bind or unbind to channel " + name + " with a null listener");
185            }
186            
187            if(eventName.startsWith(INTERNAL_EVENT_PREFIX)) {
188                throw new IllegalArgumentException("Cannot bind or unbind channel " + name + " with an internal event name such as " + eventName);
189            }
190            
191            if(state == ChannelState.UNSUBSCRIBED) {
192                throw new IllegalStateException("Cannot bind or unbind to events on a channel that has been unsubscribed. Call Pusher.subscribe() to resubscribe to this channel");
193            }
194        }
195    }
