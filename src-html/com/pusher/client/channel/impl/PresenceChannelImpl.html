001    package com.pusher.client.channel.impl;
002    
003    import java.util.Collections;
004    import java.util.HashSet;
005    import java.util.LinkedHashMap;
006    import java.util.LinkedHashSet;
007    import java.util.List;
008    import java.util.Map;
009    import java.util.Set;
010    
011    import com.google.gson.Gson;
012    import com.pusher.client.AuthorizationFailureException;
013    import com.pusher.client.Authorizer;
014    import com.pusher.client.channel.User;
015    import com.pusher.client.channel.PresenceChannel;
016    import com.pusher.client.channel.PresenceChannelEventListener;
017    import com.pusher.client.channel.SubscriptionEventListener;
018    import com.pusher.client.connection.impl.InternalConnection;
019    import com.pusher.client.util.Factory;
020    
021    public class PresenceChannelImpl extends PrivateChannelImpl implements PresenceChannel {
022    
023        private static final String MEMBER_ADDED_EVENT = "pusher_internal:member_added";
024        private static final String MEMBER_REMOVED_EVENT = "pusher_internal:member_removed";
025        private final Map<String, User> idToUserMap = Collections.synchronizedMap(new LinkedHashMap<String, User>());
026        private String myUserID;
027        
028        public PresenceChannelImpl(InternalConnection connection, String channelName, Authorizer authorizer) {
029            super(connection, channelName, authorizer);
030        }
031    
032        /* PresenceChannel implementation */
033        
034        @Override
035        public Set<User> getUsers() {
036            return new LinkedHashSet<User>(idToUserMap.values());
037        }
038        
039        @Override
040        public User getMe() {
041            return idToUserMap.get(myUserID);
042        }
043        
044        /* Base class overrides */
045        
046        @Override
047        public void onMessage(String event, String message) {
048    
049            super.onMessage(event, message);
050            
051            if(event.equals(SUBSCRIPTION_SUCCESS_EVENT)) {
052                handleSubscriptionSuccessfulMessage(message);
053            } else if(event.equals(MEMBER_ADDED_EVENT)) {
054                handleMemberAddedEvent(message);
055            } else if(event.equals(MEMBER_REMOVED_EVENT)) {
056                handleMemberRemovedEvent(message);
057            }
058        }
059    
060        @Override
061        @SuppressWarnings("rawtypes")
062        public String toSubscribeMessage() {
063            
064            String authResponse = getAuthResponse();
065            
066            try {
067            Map authResponseMap = new Gson().fromJson(authResponse, Map.class);
068            String authKey = (String) authResponseMap.get("auth");
069            Object channelData = authResponseMap.get("channel_data");
070            
071            storeMyUserId(channelData);
072            
073            Map<Object, Object> jsonObject = new LinkedHashMap<Object, Object>();
074            jsonObject.put("event", "pusher:subscribe");
075            
076            Map<Object, Object> dataMap = new LinkedHashMap<Object, Object>();
077            dataMap.put("channel", name);
078            dataMap.put("auth", authKey);
079            dataMap.put("channel_data", channelData);
080            
081            jsonObject.put("data", dataMap);
082            
083            String json = new Gson().toJson(jsonObject);
084            
085            return json;
086            } catch(Exception e) {
087                    throw new AuthorizationFailureException("Unable to parse response from Authorizer: " + authResponse, e);
088            }
089        }    
090    
091        @Override
092        public void bind(String eventName, SubscriptionEventListener listener) {
093            
094            if( (listener instanceof PresenceChannelEventListener) == false) {
095                throw new IllegalArgumentException("Only instances of PresenceChannelEventListener can be bound to a presence channel");
096            }
097            
098            super.bind(eventName, listener);
099        }
100        
101        @Override
102        protected String[] getDisallowedNameExpressions() {
103            return new String[] {
104                    "^(?!presence-).*"
105            };
106        }
107    
108        @Override
109        public String toString() {
110            return String.format("[Presence Channel: name=%s]", name);
111        }
112        
113        @SuppressWarnings({ "rawtypes", "unchecked" })
114        private void handleSubscriptionSuccessfulMessage(String message) {
115            
116            // extract data from the JSON message
117            Map presenceMap = extractPresenceMapFrom(message);
118    
119            List<String> ids = (List<String>) presenceMap.get("ids");
120            Map hash = (Map) presenceMap.get("hash");
121            
122            // build the collection of Users
123            for(String id : ids) {
124                String userData = (hash.get(id) != null) ? hash.get(id).toString() : null;
125                User user = new User(id, userData);
126                idToUserMap.put(id, user);
127            }
128            
129            // notify the event listeners
130            for(final SubscriptionEventListener eventListener : getAllEventListeners()) {
131                Factory.getEventQueue().execute(new Runnable() {
132                    public void run() {
133                        ((PresenceChannelEventListener)eventListener).onUsersInformationReceived(name, new HashSet<User>(idToUserMap.values()));
134                    }
135                });
136            }
137        }
138    
139        @SuppressWarnings("rawtypes")
140        private void handleMemberAddedEvent(String message) {
141            
142            Map dataMap = extractDataMapFrom(message);
143            String id = (String) dataMap.get("user_id");
144            String userData = (dataMap.get("user_info") != null) ? dataMap.get("user_info").toString() : null;
145            
146            final User user = new User(id, userData);
147            idToUserMap.put(id, user);
148            
149            for(final SubscriptionEventListener eventListener : getAllEventListeners()) {
150                Factory.getEventQueue().execute(new Runnable() {
151                    public void run() {
152                        ((PresenceChannelEventListener)eventListener).userSubscribed(name, user);
153                    }
154                });
155            }
156        }
157    
158        @SuppressWarnings("rawtypes")
159        private void handleMemberRemovedEvent(String message) {
160            
161            Map dataMap = extractDataMapFrom(message);
162            String id = (String) dataMap.get("user_id");
163            
164            final User user = idToUserMap.remove(id);
165            
166            for(final SubscriptionEventListener eventListener : getAllEventListeners()) {
167                Factory.getEventQueue().execute(new Runnable() {
168                    public void run() {
169                        ((PresenceChannelEventListener)eventListener).userUnsubscribed(name, user);
170                    }
171                });
172            }
173        }
174        
175        @SuppressWarnings("rawtypes")
176        private static Map extractDataMapFrom(String message) {
177            Gson gson = new Gson();
178            Map jsonObject = gson.fromJson(message, Map.class);
179            String dataString = (String) jsonObject.get("data");
180                
181            Map dataMap = gson.fromJson(dataString, Map.class);
182            return dataMap;
183        }
184        
185        @SuppressWarnings("rawtypes")
186        private static Map extractPresenceMapFrom(String message) {
187            
188            Map dataMap = extractDataMapFrom(message);
189            Map presenceMap = (Map) dataMap.get("presence");
190            
191            return presenceMap;
192        }
193        
194        @SuppressWarnings("rawtypes")
195        private void storeMyUserId(Object channelData) {
196            
197            Map channelDataMap = new Gson().fromJson(((String)channelData), Map.class);
198            myUserID = (String) channelDataMap.get("user_id");
199        }
200        
201        private Set<SubscriptionEventListener> getAllEventListeners() {
202            
203            Set<SubscriptionEventListener> allListeners = new HashSet<SubscriptionEventListener>();
204            for(Set<SubscriptionEventListener> listenersForThisEvent : eventNameToListenerMap.values()) {
205                allListeners.addAll(listenersForThisEvent);
206            }
207            return allListeners;
208        }
209    }
