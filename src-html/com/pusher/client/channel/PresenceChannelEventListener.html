001package com.pusher.client.channel;
002
003import java.util.Set;
004
005import com.pusher.client.channel.User;
006
007/**
008 * Used to listen for presence specific events as well as those defined by the
009 * {@link com.pusher.client.channel.PrivateChannelEventListener
010 * PrivateChannelEventListener} and parent interfaces.
011 */
012public interface PresenceChannelEventListener extends
013                PrivateChannelEventListener {
014
015        /**
016         * Called when the subscription has succeeded and an initial list of subscribed users has been received from Pusher.
017         * @param channelName The name of the channel the list is for.
018         * @param users The users.
019         */
020        void onUsersInformationReceived(String channelName, Set<User> users);
021
022        /**
023         * Called when a new user subscribes to the channel.
024         * @param channelName channelName The name of the channel the list is for.
025         * @param user The newly subscribed user.
026         */
027        void userSubscribed(String channelName, User user);
028
029        /**
030         * Called when an existing user unsubscribes from the channel.
031         * @param channelName The name of the channel that the user unsubscribed from.
032         * @param user The user who unsubscribed.
033         */
034        void userUnsubscribed(String channelName, User user);
035}
