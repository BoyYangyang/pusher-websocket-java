001    package com.pusher.client;
002    
003    import com.pusher.client.channel.Channel;
004    import com.pusher.client.channel.ChannelEventListener;
005    import com.pusher.client.channel.PresenceChannel;
006    import com.pusher.client.channel.PresenceChannelEventListener;
007    import com.pusher.client.channel.PrivateChannel;
008    import com.pusher.client.channel.PrivateChannelEventListener;
009    import com.pusher.client.channel.impl.ChannelManager;
010    import com.pusher.client.channel.impl.InternalChannel;
011    import com.pusher.client.channel.impl.PresenceChannelImpl;
012    import com.pusher.client.channel.impl.PrivateChannelImpl;
013    import com.pusher.client.connection.Connection;
014    import com.pusher.client.connection.ConnectionEventListener;
015    import com.pusher.client.connection.ConnectionState;
016    import com.pusher.client.connection.impl.InternalConnection;
017    import com.pusher.client.util.Factory;
018    
019    /**
020     * This class is the main entry point for accessing Pusher.
021     *
022     * <p>By creating a new {@link Pusher} instance and calling {@link Pusher.connect()} a connection to Pusher is established.</p>
023     *
024     * <p>Subscriptions for data are represented by {@link com.pusher.client.channel.Channel} objects, or subclasses thereof. Subscriptions
025     * are created by calling {@link #Pusher.subscribe(String)}, {@link #Pusher.subscribePrivate(String)}, {@link #Pusher.subscribePresence(String)} or
026     * one of the overloads.</p>
027     */
028    public class Pusher {
029    
030        private final PusherOptions pusherOptions;
031        private final InternalConnection connection;
032        private final ChannelManager channelManager;
033        private final Factory factory;
034    
035        /**
036         * <p>
037         * Creates a new instance of Pusher.
038         * </p>
039         *
040         * <p>
041         * Note that if you use this constructor you will not be able to subscribe to private or presence channels because no {@link Authorizer} has been set. If you want to use private or presence channels:
042         * <ul>
043         * <li>Create an implementation of the {@link Authorizer} interface, or use the {@link com.pusher.client.util.HttpAuthorizer} provided.</li>
044         * <li>Create an instance of {@link PusherOptions} and set the authorizer on it by calling {@link PusherOptions#setAuthorizer(Authorizer)}.</li>
045         * <li>Use the {@link #Pusher(String, PusherOptions)} constructor to create an instance of Pusher.</li>
046         * </ul>
047         * </p>
048         *
049         * <p>
050         * The {@link com.pusher.client.example.PrivateChannelExampleApp} and {@link com.pusher.client.example.PresenceChannelExampleApp} example applications show how to do this.
051         * </p>
052         *
053         * @param apiKey Your Pusher API key.
054         */
055        public Pusher(String apiKey) {
056    
057            this(apiKey, new PusherOptions());
058        }
059    
060        /**
061         * Creates a new instance of Pusher.
062         *
063         * @param apiKey Your Pusher API key.
064         * @param pusherOptions Options for the Pusher client library to use.
065         */
066        public Pusher(String apiKey, PusherOptions pusherOptions) {
067    
068            this(apiKey, pusherOptions, new Factory());
069        }
070    
071        /**
072         * Creates a new Pusher instance using the provided Factory,
073         * package level access for unit tests only.
074         */
075        Pusher(String apiKey, PusherOptions pusherOptions, Factory factory) {
076    
077            if (apiKey == null || apiKey.length() == 0) {
078                throw new IllegalArgumentException("API Key cannot be null or empty");
079            }
080    
081            if (pusherOptions == null) {
082                throw new IllegalArgumentException("PusherOptions cannot be null");
083            }
084    
085            this.pusherOptions = pusherOptions;
086            this.factory = factory;
087            this.connection = factory.getConnection(apiKey, this.pusherOptions);
088            this.channelManager = factory.getChannelManager();
089            this.channelManager.setConnection(this.connection);
090        }
091    
092        /* Connection methods */
093    
094        /**
095         * Gets the underlying {@link Connection} object that is being used by this instance of {@linkplain Pusher}.
096         *
097         * @return The {@link Connection} object.
098         */
099        public Connection getConnection() {
100            return connection;
101        }
102    
103        /**
104         * Connects to Pusher. Any {@link ConnectionEventListener}s that have already been registered using
105         * the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method will receive connection events.
106         *
107         * <p>Calls are ignored (a connection is not attempted) if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.DISCONNECTED}.</p>
108         */
109        public void connect() {
110            connect(null);
111        }
112    
113        /**
114         * Binds a {@link ConnectionEventListener} to the specified events and then connects to Pusher. This is equivalent to binding a {@link ConnectionEventListener} using the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method before connecting.
115         *
116         * <p>Calls are ignored (a connection is not attempted) if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.DISCONNECTED}.</p>
117         *
118         * @param eventListener A {@link ConnectionEventListener} that will receive connection events. This can be null if you are not interested in receiving connection events, in which case you should call {@link #connect()} instead of this method.
119         * @param connectionStates An optional list of {@link ConnectionState}s to bind your {@link ConnectionEventListener} to before connecting to Pusher. If you do not specify any {@link ConnectionState}s then your {@link ConnectionEventListener} will be bound to all connection events. This is equivalent to calling {@link #connect(ConnectionEventListener, ConnectionState...)} with {@link ConnectionState#ALL}.
120         * @throws IllegalArgumentException If the {@link ConnectionEventListener} is null and at least one connection state has been specified.
121         */
122        public void connect(ConnectionEventListener eventListener, ConnectionState... connectionStates) {
123    
124            if (eventListener != null) {
125                if (connectionStates.length == 0) {
126                    connectionStates = new ConnectionState[] { ConnectionState.ALL };
127                }
128    
129                for (ConnectionState state : connectionStates) {
130                    connection.bind(state, eventListener);
131                }
132            } else {
133                if (connectionStates.length > 0) {
134                    throw new IllegalArgumentException("Cannot bind to connection states with a null connection event listener");
135                }
136            }
137    
138            connection.connect();
139        }
140    
141        /**
142         * Disconnect from Pusher.
143         *
144         * <p>Calls are ignored if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.CONNECTED}.</p>
145         */
146        public void disconnect() {
147            if (connection.getState() == ConnectionState.CONNECTED) {
148                connection.disconnect();
149            }
150        }
151    
152        /* Subscription methods */
153    
154        /**
155         * Subscribes to a public {@link Channel}.
156         *
157         * Note that subscriptions should be registered only once with a Pusher instance.
158         * Subscriptions are persisted over disconnection and re-registered with the server automatically on reconnection.
159         * This means that subscriptions may also be registered before connect() is called, they will be initiated on connection.
160         *
161         * @param channelName The name of the {@link Channel} to subscribe to.
162         * @return The {@link Channel} object representing your subscription.
163         */
164        public Channel subscribe(String channelName) {
165            return subscribe(channelName, null);
166        }
167    
168        /**
169         * Binds a {@link ChannelEventListener} to the specified events and then subscribes to a public {@link Channel}.
170         *
171         * @param channelName The name of the {@link Channel} to subscribe to.
172         * @param listener A {@link ChannelEventListener} to receive events. This can be null if you don't want to bind a listener at subscription time, in which case you should call {@link #subscribe(String)} instead of this method.
173         * @param eventNames An optional list of event names to bind your {@link ChannelEventListener} to before subscribing.
174         * @return The {@link Channel} object representing your subscription.
175         * @throws IllegalArgumentException If any of the following are true:
176         *             <ul>
177         *             <li>The channel name is null.</li>
178         *             <li>You are already subscribed to this channel.</li>
179         *             <li>The channel name starts with "private-". If you want to subscribe to a private channel, call {@link #subscribePrivate(String, PrivateChannelEventListener, String...)} instead of this method.</li>
180         *             <li>At least one of the specified event names is null.</li>
181         *             <li>You have specified at least one event name and your {@link ChannelEventListener} is null.</li>
182         *             </ul>
183         */
184        public Channel subscribe(String channelName, ChannelEventListener listener, String... eventNames) {
185    
186            InternalChannel channel = factory.newPublicChannel(channelName);
187            channelManager.subscribeTo(channel, listener, eventNames);
188    
189            return channel;
190        }
191    
192        /**
193         * Subscribes to a {@link com.pusher.client.channel.PrivateChannel} which requires authentication.
194         *
195         * @param channelName The name of the channel to subscribe to.
196         * @return A new {@link com.pusher.client.channel.PrivateChannel} representing the subscription.
197         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
198         */
199        public PrivateChannel subscribePrivate(String channelName) {
200            return subscribePrivate(channelName, null);
201        }
202    
203        /**
204         * Subscribes to a {@link com.pusher.client.channel.PrivateChannel} which requires authentication.
205         *
206         * @param channelName The name of the channel to subscribe to.
207         * @param listener A listener to be informed of both Pusher channel protocol events and subscription data events.
208         * @param eventNames An optional list of names of events to be bound to on the channel. The equivalent of calling {@link com.pusher.client.channel.Channel.bind(String, SubscriptionListener)} on or more times.
209         * @return A new {@link com.pusher.client.channel.PrivateChannel} representing the subscription.
210         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
211         */
212        public PrivateChannel subscribePrivate(String channelName, PrivateChannelEventListener listener, String... eventNames) {
213    
214            throwExceptionIfNoAuthorizerHasBeenSet();
215    
216            PrivateChannelImpl channel = factory.newPrivateChannel(connection, channelName, pusherOptions.getAuthorizer());
217            channelManager.subscribeTo(channel, listener, eventNames);
218    
219            return channel;
220        }
221    
222        /**
223         * Subscribes to a {@link com.pusher.client.channel.PresenceChannel} which requires authentication.
224         *
225         * @param channelName The name of the channel to subscribe to.
226         * @return A new {@link com.pusher.client.channel.PresenceChannel} representing the subscription.
227         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
228         */
229        public PresenceChannel subscribePresence(String channelName) {
230            return subscribePresence(channelName, null);
231        }
232    
233        /**
234         * Subscribes to a {@link com.pusher.client.channel.PresenceChannel} which requires authentication.
235         *
236         * @param channelName The name of the channel to subscribe to.
237         * @param listener A listener to be informed of Pusher channel protocol, including presence-specific events, and subscription data events.
238         * @param eventNames An optional list of names of events to be bound to on the channel. The equivalent of calling {@link com.pusher.client.channel.Channel.bind(String, SubscriptionListener)} on or more times.
239         * @return A new {@link com.pusher.client.channel.PresenceChannel} representing the subscription.
240         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
241         */
242        public PresenceChannel subscribePresence(String channelName, PresenceChannelEventListener listener, String... eventNames) {
243    
244            throwExceptionIfNoAuthorizerHasBeenSet();
245    
246            PresenceChannelImpl channel = factory.newPresenceChannel(connection, channelName, pusherOptions.getAuthorizer());
247            channelManager.subscribeTo(channel, listener, eventNames);
248    
249            return channel;
250        }
251    
252        /**
253         * Unsubscribes from a channel using via the name of the channel.
254         * @param channelName the name of the channel to be unsubscribed from.
255         *
256         * @throws IllegalStateException if {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.CONNECTED CONNECTED}
257         */
258        public void unsubscribe(String channelName) {
259    
260            if (connection.getState() != ConnectionState.CONNECTED) {
261                throw new IllegalStateException("Cannot unsubscribe from channel " + channelName + " while not connected");
262            }
263    
264            channelManager.unsubscribeFrom(channelName);
265        }
266    
267        /* implementation detail */
268    
269        private void throwExceptionIfNoAuthorizerHasBeenSet() {
270            if (pusherOptions.getAuthorizer() == null) {
271                throw new IllegalStateException("Cannot subscribe to a private or presence channel because no Authorizer has been set. Call PusherOptions.setAuthorizer() before connecting to Pusher");
272            }
273        }
274    }
