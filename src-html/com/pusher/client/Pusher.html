001package com.pusher.client;
002
003import com.pusher.client.channel.Channel;
004import com.pusher.client.channel.ChannelEventListener;
005import com.pusher.client.channel.PresenceChannel;
006import com.pusher.client.channel.PresenceChannelEventListener;
007import com.pusher.client.channel.PrivateChannel;
008import com.pusher.client.channel.PrivateChannelEventListener;
009import com.pusher.client.channel.impl.ChannelManager;
010import com.pusher.client.channel.impl.InternalChannel;
011import com.pusher.client.channel.impl.PresenceChannelImpl;
012import com.pusher.client.channel.impl.PrivateChannelImpl;
013import com.pusher.client.connection.Connection;
014import com.pusher.client.connection.ConnectionEventListener;
015import com.pusher.client.connection.ConnectionState;
016import com.pusher.client.connection.impl.InternalConnection;
017import com.pusher.client.util.Factory;
018
019/**
020 * This class is the main entry point for accessing Pusher.
021 * 
022 * <p>By creating a new {@link Pusher} instance and calling {@link Pusher.connect()} a connection to Pusher is established.</p>
023 *
024 * <p>Subscriptions for data are represented by {@link com.pusher.client.channel.Channel} objects, or subclasses thereof. Subscriptions
025 * are created by calling {@link #Pusher.subscribe(String)}, {@link #Pusher.subscribePrivate(String)}, {@link #Pusher.subscribePresence(String)} or
026 * one of the overloads.</p>
027 */
028public class Pusher {
029
030        private final PusherOptions pusherOptions;
031        private final InternalConnection connection;
032        private final ChannelManager channelManager;
033
034        /**
035         * <p>
036         * Creates a new instance of Pusher.
037         * </p>
038         * 
039         * <p>
040         * Note that if you use this constructor you will not be able to subscribe to private or presence channels because no {@link Authorizer} has been set. If you want to use private or presence channels:
041         * <ul>
042         * <li>Create an implementation of the {@link Authorizer} interface, or use the {@link com.pusher.client.util.HttpAuthorizer} provided.</li>
043         * <li>Create an instance of {@link PusherOptions} and set the authorizer on it by calling {@link PusherOptions#setAuthorizer(Authorizer)}.</li>
044         * <li>Use the {@link #Pusher(String, PusherOptions)} constructor to create an instance of Pusher.</li>
045         * </ul>
046         * </p>
047         * 
048         * <p>
049         * The {@link com.pusher.client.example.PrivateChannelExampleApp} and {@link com.pusher.client.example.PresenceChannelExampleApp} example applications show how to do this.
050         * </p>
051         * 
052         * @param apiKey Your Pusher API key.
053         */
054        public Pusher(String apiKey) {
055
056                this(apiKey, new PusherOptions());
057        }
058
059        /**
060         * Creates a new instance of Pusher.
061         * 
062         * @param apiKey Your Pusher API key.
063         * @param pusherOptions Options for the Pusher client library to use.
064         */
065        public Pusher(String apiKey, PusherOptions pusherOptions) {
066                
067                if (apiKey == null || apiKey.length() == 0) {
068                        throw new IllegalArgumentException("API Key cannot be null or empty");
069                }
070
071                if (pusherOptions == null) {
072                        throw new IllegalArgumentException("PusherOptions cannot be null");
073                }
074
075                this.pusherOptions = pusherOptions;
076                this.connection = Factory.getConnection(apiKey, this.pusherOptions.isEncrypted());
077                this.channelManager = Factory.getChannelManager();
078                this.channelManager.setConnection(this.connection);
079        }
080
081        /* Connection methods */
082
083        /**
084         * Gets the underlying {@link Connection} object that is being used by this instance of {@linkplain Pusher}.
085         * 
086         * @return The {@link Connection} object.
087         */
088        public Connection getConnection() {
089                return connection;
090        }
091
092        /**
093         * Connects to Pusher. Any {@link ConnectionEventListener}s that have already been registered using
094         * the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method will receive connection events.
095         * 
096         * <p>Calls are ignored (a connection is not attempted) if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.DISCONNECTED}.</p>
097         */
098        public void connect() {
099                connect(null);
100        }
101
102        /**
103         * Binds a {@link ConnectionEventListener} to the specified events and then connects to Pusher. This is equivalent to binding a {@link ConnectionEventListener} using the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method before connecting.
104         * 
105         * <p>Calls are ignored (a connection is not attempted) if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.DISCONNECTED}.</p>
106         * 
107         * @param eventListener A {@link ConnectionEventListener} that will receive connection events. This can be null if you are not interested in receiving connection events, in which case you should call {@link #connect()} instead of this method.
108         * @param connectionStates An optional list of {@link ConnectionState}s to bind your {@link ConnectionEventListener} to before connecting to Pusher. If you do not specify any {@link ConnectionState}s then your {@link ConnectionEventListener} will be bound to all connection events. This is equivalent to calling {@link #connect(ConnectionEventListener, ConnectionState...)} with {@link ConnectionState#ALL}.
109         * @throws IllegalArgumentException If the {@link ConnectionEventListener} is null and at least one connection state has been specified.
110         */
111        public void connect(ConnectionEventListener eventListener, ConnectionState... connectionStates) {
112
113                if (eventListener != null) {
114                        if (connectionStates.length == 0) {
115                                connectionStates = new ConnectionState[] { ConnectionState.ALL };
116                        }
117
118                        for (ConnectionState state : connectionStates) {
119                                connection.bind(state, eventListener);
120                        }
121                } else {
122                        if (connectionStates.length > 0) {
123                                throw new IllegalArgumentException("Cannot bind to connection states with a null connection event listener");
124                        }
125                }
126
127                connection.connect();
128        }
129
130        /**
131         * Disconnect from Pusher.
132         * 
133         * <p>Calls are ignored if the {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.CONNECTED}.</p>
134         */
135        public void disconnect() {
136                if (connection.getState() == ConnectionState.CONNECTED) {
137                        connection.disconnect();
138                }
139        }
140
141        /* Subscription methods */
142
143        /**
144         * Subscribes to a public {@link Channel}.
145         * 
146         * @param channelName The name of the {@link Channel} to subscribe to.
147         * @return The {@link Channel} object representing your subscription.
148         */
149        public Channel subscribe(String channelName) {
150                return subscribe(channelName, null);
151        }
152
153        /**
154         * Binds a {@link ChannelEventListener} to the specified events and then subscribes to a public {@link Channel}.
155         * 
156         * @param channelName The name of the {@link Channel} to subscribe to.
157         * @param listener A {@link ChannelEventListener} to receive events. This can be null if you don't want to bind a listener at subscription time, in which case you should call {@link #subscribe(String)} instead of this method.
158         * @param eventNames An optional list of event names to bind your {@link ChannelEventListener} to before subscribing.
159         * @return The {@link Channel} object representing your subscription.
160         * @throws IllegalArgumentException If any of the following are true:
161         *             <ul>
162         *             <li>The channel name is null.</li>
163         *             <li>You are already subscribed to this channel.</li>
164         *             <li>The channel name starts with "private-". If you want to subscribe to a private channel, call {@link #subscribePrivate(String, PrivateChannelEventListener, String...)} instead of this method.</li>
165         *             <li>At least one of the specified event names is null.</li>
166         *             <li>You have specified at least one event name and your {@link ChannelEventListener} is null.</li>
167         *             </ul>
168         */
169        public Channel subscribe(String channelName, ChannelEventListener listener, String... eventNames) {
170
171                InternalChannel channel = Factory.newPublicChannel(channelName);
172                channelManager.subscribeTo(channel, listener, eventNames);
173
174                return channel;
175        }
176
177        /**
178         * Subscribes to a {@link com.pusher.client.channel.PrivateChannel} which requires authentication.
179         * 
180         * @param channelName The name of the channel to subscribe to.
181         * @return A new {@link com.pusher.client.channel.PrivateChannel} representing the subscription.
182         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
183         */
184        public PrivateChannel subscribePrivate(String channelName) {
185                return subscribePrivate(channelName, null);
186        }
187
188        /**
189         * Subscribes to a {@link com.pusher.client.channel.PrivateChannel} which requires authentication.
190         * 
191         * @param channelName The name of the channel to subscribe to.
192         * @param listener A listener to be informed of both Pusher channel protocol events and subscription data events.
193         * @param eventNames An optional list of names of events to be bound to on the channel. The equivalent of calling {@link com.pusher.client.channel.Channel.bind(String, SubscriptionListener)} on or more times.
194         * @return A new {@link com.pusher.client.channel.PrivateChannel} representing the subscription.
195         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
196         */
197        public PrivateChannel subscribePrivate(String channelName, PrivateChannelEventListener listener, String... eventNames) {
198
199                throwExceptionIfNoAuthorizerHasBeenSet();
200
201                PrivateChannelImpl channel = Factory.newPrivateChannel(connection, channelName, pusherOptions.getAuthorizer());
202                channelManager.subscribeTo(channel, listener, eventNames);
203
204                return channel;
205        }
206
207        /**
208         * Subscribes to a {@link com.pusher.client.channel.PresenceChannel} which requires authentication.
209         * 
210         * @param channelName The name of the channel to subscribe to.
211         * @return A new {@link com.pusher.client.channel.PresenceChannel} representing the subscription.
212         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
213         */
214        public PresenceChannel subscribePresence(String channelName) {
215                return subscribePresence(channelName, null);
216        }
217
218        /**
219         * Subscribes to a {@link com.pusher.client.channel.PresenceChannel} which requires authentication.
220         * 
221         * @param channelName The name of the channel to subscribe to.
222         * @param listener A listener to be informed of Pusher channel protocol, including presence-specific events, and subscription data events.
223         * @param eventNames An optional list of names of events to be bound to on the channel. The equivalent of calling {@link com.pusher.client.channel.Channel.bind(String, SubscriptionListener)} on or more times.
224         * @return A new {@link com.pusher.client.channel.PresenceChannel} representing the subscription.
225         * @throws IllegalStateException if a {@link com.pusher.client.Authorizer} has not been set for the {@link Pusher} instance via {@link #Pusher(String, PusherOptions)}.
226         */
227        public PresenceChannel subscribePresence(String channelName, PresenceChannelEventListener listener, String... eventNames) {
228
229                throwExceptionIfNoAuthorizerHasBeenSet();
230
231                PresenceChannelImpl channel = Factory.newPresenceChannel(connection, channelName, pusherOptions.getAuthorizer());
232                channelManager.subscribeTo(channel, listener, eventNames);
233
234                return channel;
235        }
236
237        /**
238         * Unsubscribes from a channel using via the name of the channel.
239         * @param channelName the name of the channel to be unsubscribed from.
240         * 
241         * @throws IllegalStateException if {@link Pusher.getConnection().getState()} is not {@link com.pusher.client.connection.ConnectionState.CONNECTED CONNECTED}
242         */
243        public void unsubscribe(String channelName) {
244
245                if (connection.getState() != ConnectionState.CONNECTED) {
246                        throw new IllegalStateException("Cannot unsubscribe from channel " + channelName + " while not connected");
247                }
248
249                channelManager.unsubscribeFrom(channelName);
250        }
251
252        /* implementation detail */
253
254        private void throwExceptionIfNoAuthorizerHasBeenSet() {
255                if (pusherOptions.getAuthorizer() == null) {
256                        throw new IllegalStateException("Cannot subscribe to a private or presence channel because no Authorizer has been set. Call PusherOptions.setAuthorizer() before connecting to Pusher");
257                }
258        }
259}
