001    package com.pusher.client.example;
002    
003    import java.net.URI;
004    import java.net.URISyntaxException;
005    
006    import org.java_websocket.client.WebSocketClient;
007    import org.java_websocket.handshake.ServerHandshake;
008    
009    public class SimpleWebSocket extends WebSocketClient {
010            public static void main(String[] args) throws URISyntaxException {
011                    new SimpleWebSocket();
012            }
013            
014            public SimpleWebSocket() throws URISyntaxException {
015                    super(new URI("ws://ws.pusherapp.com/app/387954142406c3c9cc13?protocol=6&client=js&version=0.1.2&flash=false"));
016                    
017                    System.out.println("SimpleWebSocket");
018                    
019                    this.connect();
020            }
021    
022            @Override
023            public void onOpen(ServerHandshake handshakedata) {
024                    System.out.println("onOpen");
025                    
026            }
027    
028            @Override
029            public void onMessage(String message) {
030                    System.out.println("onMessage: " + message);
031            }
032    
033            @Override
034            public void onClose(int code, String reason, boolean remote) {
035                    System.out.println("onClose");
036            }
037    
038            @Override
039            public void onError(Exception ex) {
040                    System.out.println("onError");
041            }
042    }
