001    package com.pusher.client.example;
002    
003    import com.pusher.client.Pusher;
004    import com.pusher.client.PusherOptions;
005    import com.pusher.client.channel.PrivateChannel;
006    import com.pusher.client.channel.PrivateChannelEventListener;
007    import com.pusher.client.connection.ConnectionEventListener;
008    import com.pusher.client.connection.ConnectionStateChange;
009    import com.pusher.client.util.HttpAuthorizer;
010    
011    public class PrivateChannelExampleApp implements ConnectionEventListener,
012                    PrivateChannelEventListener {
013    
014            private final Pusher pusher;
015            private final String channelName;
016            private final String eventName;
017    
018            private PrivateChannel channel;
019    
020            public static void main(String[] args) {
021                    new PrivateChannelExampleApp(args);
022            }
023    
024            public PrivateChannelExampleApp(String[] args) {
025    
026                    String apiKey = (args.length > 0) ? args[0] : "a87fe72c6f36272aa4b1";
027                    channelName = (args.length > 1) ? args[1] : "private-my-channel";
028                    eventName = (args.length > 2) ? args[2] : "my-event";
029    
030                    HttpAuthorizer authorizer = new HttpAuthorizer(
031                                    "http://www.leggetter.co.uk/pusher/pusher-examples/php/authentication/src/private_auth.php");
032                    PusherOptions options = new PusherOptions().setAuthorizer(authorizer);
033    
034                    pusher = new Pusher(apiKey, options);
035                    pusher.connect(this);
036                    
037                    channel = pusher.subscribePrivate(channelName, this, eventName);
038            }
039    
040            /* ConnectionEventListener implementation */
041    
042            @Override
043            public void onConnectionStateChange(ConnectionStateChange change) {
044    
045                    System.out.println(String.format(
046                                    "Connection state changed from [%s] to [%s]",
047                                    change.getPreviousState(), change.getCurrentState()));
048            }
049    
050            @Override
051            public void onError(String message, String code, Exception e) {
052    
053                    System.out.println(String.format(
054                                    "An error was received with message [%s], code [%s], exception [%s]",
055                                    message, code, e));
056            }
057    
058            /* PrivateChannelEventListener implementation */
059    
060            @Override
061            public void onEvent(String channelName, String eventName, String data) {
062    
063                    System.out.println(String.format(
064                                    "Received event [%s] on channel [%s] with data [%s]", eventName,
065                                    channelName, data));
066            }
067    
068            @Override
069            public void onSubscriptionSucceeded(String channelName) {
070    
071                    System.out.println(String.format("Subscription to channel [%s] succeeded",
072                                    channel.getName()));
073    
074                    channel.trigger("client-myEvent", "{\"myName\":\"Bob\"}");
075            }
076    
077            @Override
078            public void onAuthenticationFailure(String message, Exception e) {
079    
080                    System.out.println(String.format(
081                                    "Authentication failure due to [%s], exception was [%s]", message, e));
082            }
083    }
