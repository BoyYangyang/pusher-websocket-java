001package com.pusher.client.connection;
002
003/**
004 * <p>Client applications should implement this interface if they wish to receive notifications when the state of a
005 * {@link Connection} changes or an error is thrown.</p>
006 * 
007 * <p>Implementations of this interface can be bound to the connection by calling {@link Connection#bind(ConnectionState, ConnectionEventListener)}. 
008 * The connection itself can be retrieved from the {@link com.pusher.client.Pusher} object by calling 
009 * {@link com.pusher.client.Pusher#getConnection()}.</p>
010 * 
011 * <p>Alternatively, you can bind your implementation of the interface and connect at the same time by calling 
012 * {@link com.pusher.client.Pusher#connect(ConnectionEventListener, ConnectionState...)}.</p>
013 */
014public interface ConnectionEventListener {
015
016    /**
017     * <p>Callback that is fired whenever the {@link ConnectionState} of the {@link Connection} changes. The state 
018     * typically changes during connection to Pusher and during disconnection and reconnection.</p>
019     * 
020     * <p>This callback is only fired if the {@linkplain ConnectionEventListener} has been bound to the new state by
021     * calling {@link Connection#bind(ConnectionState, ConnectionEventListener)} with either the new state or
022     * {@link ConnectionState#ALL}.</p>
023     * @param change An object that contains the previous state of the connection and the new state. The new state
024     * can be retrieved by calling {@link ConnectionStateChange#getCurrentState()}.
025     */
026    void onConnectionStateChange(ConnectionStateChange change);
027
028    /**
029     * <p>Callback that indicates either:
030     * 
031     * <ul>
032     *  <li>An error message has been received from Pusher, or</li>
033     *  <li>An error has occurred in the client library.</li>
034     * </ul></p>
035     * 
036     * <p>All {@linkplain ConnectionEventListener}s that have been registered by calling {@link Connection#bind(ConnectionState, ConnectionEventListener)} 
037     * will receive this callback, even if the {@linkplain ConnectionEventListener} is only bound to specific 
038     * connection status changes.</p>
039     * 
040     * @param message A message indicating the cause of the error.
041     * @param code The error code for the message. Can be null.
042     * @param e The exception that was thrown, if any. Can be null.
043     */
044    void onError(String message, String code, Exception e);
045}
