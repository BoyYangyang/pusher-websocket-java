001    package com.pusher.client.connection.websocket;
002    
003    import java.net.URI;
004    import java.security.KeyManagementException;
005    import java.security.NoSuchAlgorithmException;
006    
007    import javax.net.ssl.SSLContext;
008    import javax.net.ssl.SSLException;
009    
010    import org.java_websocket.client.DefaultSSLWebSocketClientFactory;
011    import org.java_websocket.client.WebSocketClient;
012    import org.java_websocket.handshake.ServerHandshake;
013    
014    /**
015     * A thin wrapper around the WebSocketClient class from the Java-WebSocket library. The purpose of this
016     * class is to enable the WebSocketConnection class to be unit tested by swapping out an instance of
017     * this wrapper for a mock version. 
018     */
019    public class WebSocketClientWrapper extends WebSocketClient {
020    
021            private static final String WSS_SCHEME = "wss";
022            private final WebSocketListener proxy;
023        
024            public WebSocketClientWrapper(URI uri, WebSocketListener proxy)
025                            throws SSLException {
026                    super(uri);
027    
028                    if (uri.getScheme().equals( WSS_SCHEME )) {
029                            try {
030                                    SSLContext sslContext = SSLContext.getInstance("TLS");
031                                    sslContext.init(null, null, null);
032    
033                                    this.setWebSocketFactory(new DefaultSSLWebSocketClientFactory(
034                                                    sslContext));
035                            }
036                            catch (NoSuchAlgorithmException e) {
037                                    throw new SSLException(e);
038                            }
039                            catch (KeyManagementException e) {
040                                    throw new SSLException(e);
041                            }
042                    }
043    
044                    this.proxy = proxy;
045            }
046        
047        @Override
048        public void onOpen(ServerHandshake handshakedata) {
049            proxy.onOpen(handshakedata);
050        }
051    
052        @Override
053        public void onMessage(String message) {
054            proxy.onMessage(message);
055        }
056    
057        @Override
058        public void onClose(int code, String reason, boolean remote) {
059            proxy.onClose(code, reason, remote);
060        }
061    
062        @Override
063        public void onError(Exception ex) {
064            proxy.onError(ex);
065        }    
066    }
