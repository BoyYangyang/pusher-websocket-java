001package com.pusher.client.connection;
002
003/**
004 * Represents a change in connection state.
005 */
006public class ConnectionStateChange {
007
008        private final ConnectionState previousState;
009        private final ConnectionState currentState;
010
011        /**
012         * Used within the library to create a connection state change. Not be used used as part of the API.
013         * @param previousState
014         * @param currentState
015         */
016        public ConnectionStateChange(ConnectionState previousState,
017                        ConnectionState currentState) {
018
019                if (previousState == currentState) {
020                        throw new IllegalArgumentException(
021                                        "Attempted to create an connection state update where both previous and current state are: "
022                                                        + currentState);
023                }
024
025                this.previousState = previousState;
026                this.currentState = currentState;
027        }
028
029        /**
030         * The previous connections state. The state the connection has transitioned from.
031         * @return
032         */
033        public ConnectionState getPreviousState() {
034                return previousState;
035        }
036
037        /**
038         * The current connection state. The state the connection has transitioned to.
039         * @return
040         */
041        public ConnectionState getCurrentState() {
042                return currentState;
043        }
044
045        @Override
046        public int hashCode() {
047                return previousState.hashCode() + currentState.hashCode();
048        }
049
050        @Override
051        public boolean equals(Object obj) {
052                if (obj != null && obj instanceof ConnectionStateChange) {
053                        ConnectionStateChange other = (ConnectionStateChange) obj;
054                        return (this.currentState == other.currentState)
055                                        && (this.previousState == other.previousState);
056                }
057
058                return false;
059        }
060}
