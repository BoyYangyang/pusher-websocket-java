001    package com.pusher.client.connection;
002    
003    /**
004     * Represents a connection to Pusher.
005     * 
006     */
007    public interface Connection {
008    
009            /**
010             * No need to call this via the API. Instead use {@link com.pusher.client.Pusher.connect()}.
011             */
012            void connect();
013    
014            /**
015             * Bind to connection events.
016             * @param state The states to bind to.
017             * @param eventListener A listener to be called when the state changes.
018             */
019            void bind(ConnectionState state, ConnectionEventListener eventListener);
020    
021            /**
022             * Unbind from connection state changes.
023             * @param state The state to unbind from.
024             * @param eventListener The listener to be unbound.
025             * @return <code>true</code> if the unbind was successful, otherwise <code>false</code>.
026             */
027            boolean unbind(ConnectionState state, ConnectionEventListener eventListener);
028    
029            /**
030             * Gets the current connection state.
031             * @return The state.
032             */
033            ConnectionState getState();
034    
035            /**
036             * Gets a unique connection ID.
037             * @return The id.
038             */
039            String getSocketId();
040    }
