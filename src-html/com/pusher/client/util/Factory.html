001package com.pusher.client.util;
002
003import java.net.MalformedURLException;
004import java.net.URI;
005import java.net.URISyntaxException;
006import java.net.URL;
007import java.util.concurrent.ExecutorService;
008import java.util.concurrent.Executors;
009
010import javax.net.ssl.SSLException;
011
012import org.java_websocket.client.WebSocketClient;
013
014import com.pusher.client.Authorizer;
015import com.pusher.client.channel.impl.ChannelImpl;
016import com.pusher.client.channel.impl.ChannelManager;
017import com.pusher.client.channel.impl.PresenceChannelImpl;
018import com.pusher.client.channel.impl.PrivateChannelImpl;
019import com.pusher.client.connection.impl.InternalConnection;
020import com.pusher.client.connection.websocket.WebSocketClientWrapper;
021import com.pusher.client.connection.websocket.WebSocketConnection;
022import com.pusher.client.connection.websocket.WebSocketListener;
023
024/**
025 * This is a lightweight way of doing dependency injection and enabling classes
026 * to be unit tested in isolation. No class in this library instantiates another
027 * class directly, otherwise they would be tightly coupled. Instead, they all
028 * call the factory methods in this class when they want to create instances of
029 * another class.
030 * 
031 * When unit testing we can use PowerMock to mock out the methods in this class
032 * to return mocks instead of the actual implementations. This allows us to test
033 * classes in isolation.
034 * 
035 * Conventions:
036 * 
037 * - any method that starts with "new", such as
038 * {@link #newPublicChannel(String)} creates a new instance of that class every
039 * time it is called.
040 * 
041 * - any method that starts with "get", such as {@link #getEventQueue()} returns
042 * a singleton.
043 */
044public class Factory {
045
046    private static InternalConnection connection;
047    private static ChannelManager channelManager;
048    private static ExecutorService eventQueue;
049
050    public static InternalConnection getConnection(String apiKey, boolean encrypted) {
051        if (connection == null) {
052            try {
053                connection = new WebSocketConnection(apiKey, encrypted);
054            } catch (URISyntaxException e) {
055                throw new IllegalArgumentException(
056                        "Failed to initialise connection", e);
057            }
058        }
059        return connection;
060    }
061
062    public static WebSocketClient newWebSocketClientWrapper(URI uri,
063            WebSocketListener proxy) throws SSLException {
064        return new WebSocketClientWrapper(uri, proxy);
065    }
066
067    public static ExecutorService getEventQueue() {
068        if (eventQueue == null) {
069            eventQueue = Executors.newSingleThreadExecutor();
070        }
071        return eventQueue;
072    }
073
074    public static ChannelImpl newPublicChannel(String channelName) {
075        return new ChannelImpl(channelName);
076    }
077    
078    public static PrivateChannelImpl newPrivateChannel(InternalConnection connection, String channelName, Authorizer authorizer) {
079        return new PrivateChannelImpl(connection, channelName, authorizer);
080    }
081    
082    public static PresenceChannelImpl newPresenceChannel(InternalConnection connection, String channelName, Authorizer authorizer) {
083        return new PresenceChannelImpl(connection, channelName, authorizer);
084    }
085
086    public static ChannelManager getChannelManager() {
087        if (channelManager == null) {
088            channelManager = new ChannelManager();
089        }
090        return channelManager;
091    }
092
093    public static URL newURL(String endPoint) throws MalformedURLException {
094        return new URL(endPoint);
095    }
096}
