001    package com.pusher.client.util;
002    
003    import java.net.URI;
004    import java.net.URISyntaxException;
005    import java.util.concurrent.Executors;
006    import java.util.concurrent.ScheduledExecutorService;
007    import java.util.concurrent.ThreadFactory;
008    
009    import javax.net.ssl.SSLException;
010    
011    import org.java_websocket.client.WebSocketClient;
012    
013    import com.pusher.client.Authorizer;
014    import com.pusher.client.PusherOptions;
015    import com.pusher.client.channel.impl.ChannelImpl;
016    import com.pusher.client.channel.impl.ChannelManager;
017    import com.pusher.client.channel.impl.PresenceChannelImpl;
018    import com.pusher.client.channel.impl.PrivateChannelImpl;
019    import com.pusher.client.connection.impl.InternalConnection;
020    import com.pusher.client.connection.websocket.WebSocketClientWrapper;
021    import com.pusher.client.connection.websocket.WebSocketConnection;
022    import com.pusher.client.connection.websocket.WebSocketListener;
023    
024    /**
025     * This is a lightweight way of doing dependency injection and enabling classes
026     * to be unit tested in isolation. No class in this library instantiates another
027     * class directly, otherwise they would be tightly coupled. Instead, they all
028     * call the factory methods in this class when they want to create instances of
029     * another class.
030     *
031     * An instance of Factory is provided on construction to each class which may
032     * require it, the initial factory is instantiated in the Pusher constructor,
033     * the only constructor which a library consumer should need to call directly.
034     *
035     * Conventions:
036     *
037     * - any method that starts with "new", such as
038     * {@link #newPublicChannel(String)} creates a new instance of that class every
039     * time it is called.
040     *
041     * - any method that starts with "get", such as {@link #getEventQueue()} returns
042     * a singleton. These are lazily constructed and their access methods should be
043     * synchronized for this reason.
044     */
045    public class Factory {
046    
047        private InternalConnection connection;
048        private ChannelManager channelManager;
049        private ScheduledExecutorService eventQueue;
050    
051        public synchronized InternalConnection getConnection(String apiKey, PusherOptions options) {
052            if (connection == null) {
053                try {
054                    connection = new WebSocketConnection(options.buildUrl(apiKey),
055                                                         options.getActivityTimeout(),
056                                                         options.getPongTimeout(),
057                                                         this);
058                } catch (URISyntaxException e) {
059                    throw new IllegalArgumentException("Failed to initialise connection", e);
060                }
061            }
062            return connection;
063        }
064    
065        public WebSocketClient newWebSocketClientWrapper(URI uri, WebSocketListener proxy) throws SSLException {
066            return new WebSocketClientWrapper(uri, proxy);
067        }
068    
069        public synchronized ScheduledExecutorService getEventQueue() {
070            if (eventQueue == null) {
071                eventQueue = Executors.newSingleThreadScheduledExecutor(new ThreadFactory() {
072                    @Override
073                    public Thread newThread(Runnable r) {
074                        Thread t = new Thread(r);
075                        t.setDaemon(true);
076                        t.setName("PusherClient EventQueue");
077                        return t;
078                    }
079                });
080            }
081            return eventQueue;
082        }
083    
084        public synchronized void shutdownEventQueue() {
085            if (eventQueue != null) {
086                eventQueue.shutdown();
087                eventQueue = null;
088            }
089        }
090    
091        public ChannelImpl newPublicChannel(String channelName) {
092            return new ChannelImpl(channelName, this);
093        }
094    
095        public PrivateChannelImpl newPrivateChannel(InternalConnection connection, String channelName, Authorizer authorizer) {
096            return new PrivateChannelImpl(connection, channelName, authorizer, this);
097        }
098    
099        public PresenceChannelImpl newPresenceChannel(InternalConnection connection, String channelName,
100                Authorizer authorizer) {
101            return new PresenceChannelImpl(connection, channelName, authorizer, this);
102        }
103    
104        public synchronized ChannelManager getChannelManager() {
105            if (channelManager == null) {
106                channelManager = new ChannelManager(this);
107            }
108            return channelManager;
109        }
110    
111        public long timeNow() {
112            return System.currentTimeMillis();
113        }
114    }
