001    package com.pusher.client.util;
002    
003    import java.io.BufferedReader;
004    import java.io.DataOutputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    import java.io.InputStreamReader;
008    import java.net.HttpURLConnection;
009    import java.net.MalformedURLException;
010    import java.net.URL;
011    import java.net.URLEncoder;
012    import java.util.HashMap;
013    import com.pusher.client.AuthorizationFailureException;
014    import com.pusher.client.Authorizer;
015    
016    /**
017     * Used to authenticate a {@link com.pusher.client.channel.PrivateChannel private} or {@link com.pusher.client.channel.PresenceChannel presence}
018     * channel subscription.
019     * 
020     * <p>Makes an HTTP request to a defined HTTP endpoint. Expects an authentication token to be returned.</p>
021     * 
022     * <p>For more information see the <a href="http://pusher.com/docs/authenticating_users">Authenticating Users documentation</a>.
023     */
024    public class HttpAuthorizer implements Authorizer {
025    
026            private final URL endPoint;
027            private HashMap<String, String> mHeaders = new HashMap<String, String>();
028    
029            /**
030             * Creates a new authorizer.
031             * @param endPoint The endpoint to be called when authenticating.
032             */
033            public HttpAuthorizer(String endPoint) {
034                    try {
035                            this.endPoint = Factory.newURL(endPoint);
036                    } catch (MalformedURLException e) {
037                            throw new IllegalArgumentException(
038                                            "Could not parse authentication end point into a valid URL", e);
039                    }
040            }
041            
042            /**
043             * Set additional headers to be sent as part of the request.
044             * @param headers
045             */
046            public void setHeaders(HashMap<String, String> headers) {
047                    this.mHeaders = headers;
048            }
049    
050            @Override
051            public String authorize(String channelName, String socketId)
052                            throws AuthorizationFailureException {
053    
054                    try {
055                            String urlParameters = "channel_name="
056                                            + URLEncoder.encode(channelName, "UTF-8") + "&socket_id="
057                                            + URLEncoder.encode(socketId, "UTF-8");
058    
059                            HttpURLConnection connection = (HttpURLConnection) endPoint
060                                            .openConnection();
061                            connection.setDoOutput(true);
062                            connection.setDoInput(true);
063                            connection.setInstanceFollowRedirects(false);
064                            connection.setRequestMethod("POST");
065                            connection.setRequestProperty("Content-Type",
066                                            "application/x-www-form-urlencoded");
067                            connection.setRequestProperty("charset", "utf-8");
068                            connection.setRequestProperty("Content-Length",
069                                            "" + Integer.toString(urlParameters.getBytes().length));
070    
071                            // Add in the user defined headers
072                            for (String headerName : mHeaders.keySet()) {
073                                    String headerValue = mHeaders.get(headerName);
074                                    connection.setRequestProperty(headerName, headerValue);
075                            }
076    
077                            connection.setUseCaches(false);
078    
079                            // Send request
080                            DataOutputStream wr = new DataOutputStream(connection.getOutputStream());
081                            wr.writeBytes(urlParameters);
082                            wr.flush();
083                            wr.close();
084    
085                            // Read response
086                            InputStream is = connection.getInputStream();
087                            BufferedReader rd = new BufferedReader(new InputStreamReader(is));
088                            String line;
089                            StringBuffer response = new StringBuffer();
090                            while ((line = rd.readLine()) != null) {
091                                    response.append(line);
092                            }
093                            rd.close();
094    
095                            return response.toString();
096    
097                    } catch (IOException e) {
098                            throw new AuthorizationFailureException(e);
099                    }
100            }
101    }
