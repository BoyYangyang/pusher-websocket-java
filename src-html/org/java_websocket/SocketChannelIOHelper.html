001    package org.java_websocket;
002    
003    import java.io.IOException;
004    import java.nio.ByteBuffer;
005    import java.nio.channels.ByteChannel;
006    import java.nio.channels.spi.AbstractSelectableChannel;
007    
008    public class SocketChannelIOHelper {
009    
010            public static boolean read( final ByteBuffer buf, WebSocketImpl ws, ByteChannel channel ) throws IOException {
011                    buf.clear();
012                    int read = channel.read( buf );
013                    buf.flip();
014    
015                    if( read == -1 ) {
016                            ws.eot();
017                            return false;
018                    }
019                    return read != 0;
020            }
021    
022            /**
023             * @see WrappedByteChannel#readMore(ByteBuffer)
024             * @return returns whether there is more data left which can be obtained via {@link #readMore(ByteBuffer, WebSocketImpl, WrappedByteChannel)}
025             **/
026            public static boolean readMore( final ByteBuffer buf, WebSocketImpl ws, WrappedByteChannel channel ) throws IOException {
027                    buf.clear();
028                    int read = channel.readMore( buf );
029                    buf.flip();
030    
031                    if( read == -1 ) {
032                            ws.eot();
033                            return false;
034                    }
035                    return channel.isNeedRead();
036            }
037    
038            /** Returns whether the whole outQueue has been flushed */
039            public static boolean batch( WebSocketImpl ws, ByteChannel sockchannel ) throws IOException {
040                    ByteBuffer buffer = ws.outQueue.peek();
041                    WrappedByteChannel c = null;
042    
043                    if( buffer == null ) {
044                            if( sockchannel instanceof WrappedByteChannel ) {
045                                    c = (WrappedByteChannel) sockchannel;
046                                    if( c.isNeedWrite() ) {
047                                            c.writeMore();
048                                    }
049                            }
050                    } else {
051                            do {// FIXME writing as much as possible is unfair!!
052                                    /*int written = */sockchannel.write( buffer );
053                                    if( buffer.remaining() > 0 ) {
054                                            return false;
055                                    } else {
056                                            ws.outQueue.poll(); // Buffer finished. Remove it.
057                                            buffer = ws.outQueue.peek();
058                                    }
059                            } while ( buffer != null );
060                    }
061    
062                    if( ws.outQueue.isEmpty() && ws.isFlushAndClose() /*&& ( c == null || c.isNeedWrite() )*/) {
063                            synchronized ( ws ) {
064                                    ws.closeConnection();
065                            }
066                    }
067                    return c != null ? !( (WrappedByteChannel) sockchannel ).isNeedWrite() : true;
068            }
069    
070            public static void writeBlocking( WebSocketImpl ws, ByteChannel channel ) throws InterruptedException , IOException {
071                    assert ( channel instanceof AbstractSelectableChannel == true ? ( (AbstractSelectableChannel) channel ).isBlocking() : true );
072                    assert ( channel instanceof WrappedByteChannel == true ? ( (WrappedByteChannel) channel ).isBlocking() : true );
073    
074                    ByteBuffer buf = ws.outQueue.take();
075                    while ( buf.hasRemaining() )
076                            channel.write( buf );
077            }
078    
079    }
