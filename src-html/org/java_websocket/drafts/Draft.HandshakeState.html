049            public static ByteBuffer readLine( ByteBuffer buf ) {
050                    ByteBuffer sbuf = ByteBuffer.allocate( buf.remaining() );
051                    byte prev = '0';
052                    byte cur = '0';
053                    while ( buf.hasRemaining() ) {
054                            prev = cur;
055                            cur = buf.get();
056                            sbuf.put( cur );
057                            if( prev == (byte) '\r' && cur == (byte) '\n' ) {
058                                    sbuf.limit( sbuf.position() - 2 );
059                                    sbuf.position( 0 );
060                                    return sbuf;
061    
062                            }
063                    }
064                    // ensure that there wont be any bytes skipped
065                    buf.position( buf.position() - sbuf.position() );
066                    return null;
067            }
068    
069            public static String readStringLine( ByteBuffer buf ) {
070                    ByteBuffer b = readLine( buf );
071                    return b == null ? null : Charsetfunctions.stringAscii( b.array(), 0, b.limit() );
072            }

074            public static HandshakeBuilder translateHandshakeHttp( ByteBuffer buf, Role role ) throws InvalidHandshakeException , IncompleteHandshakeException {
075                    HandshakeBuilder handshake;
076    
077                    String line = readStringLine( buf );
078                    if( line == null )
079                            throw new IncompleteHandshakeException( buf.capacity() + 128 );
080    
081                    String[] firstLineTokens = line.split( " ", 3 );// eg. HTTP/1.1 101 Switching the Protocols
082                    if( firstLineTokens.length != 3 ) {
083                            throw new InvalidHandshakeException();
084                    }
085    
086                    if( role == Role.CLIENT ) {
087                            // translating/parsing the response from the SERVER
088                            handshake = new HandshakeImpl1Server();
089                            ServerHandshakeBuilder serverhandshake = (ServerHandshakeBuilder) handshake;
090                            serverhandshake.setHttpStatus( Short.parseShort( firstLineTokens[ 1 ] ) );
091                            serverhandshake.setHttpStatusMessage( firstLineTokens[ 2 ] );
092                    } else {
093                            // translating/parsing the request from the CLIENT
094                            ClientHandshakeBuilder clienthandshake = new HandshakeImpl1Client();
095                            clienthandshake.setResourceDescriptor( firstLineTokens[ 1 ] );
096                            handshake = clienthandshake;
097                    }
098    
099                    line = readStringLine( buf );
100                    while ( line != null && line.length() > 0 ) {
101                            String[] pair = line.split( ":", 2 );
102                            if( pair.length != 2 )
103                                    throw new InvalidHandshakeException( "not an http header" );
104                            handshake.put( pair[ 0 ], pair[ 1 ].replaceFirst( "^ +", "" ) );
105                            line = readStringLine( buf );
106                    }
107                    if( line == null )
108                            throw new IncompleteHandshakeException();
109                    return handshake;
110            }

116            protected boolean basicAccept( Handshakedata handshakedata ) {
117                    return handshakedata.getFieldValue( "Upgrade" ).equalsIgnoreCase( "websocket" ) && handshakedata.getFieldValue( "Connection" ).toLowerCase( Locale.ENGLISH ).contains( "upgrade" );

132            public List<ByteBuffer> createHandshake( Handshakedata handshakedata, Role ownrole, boolean withcontent ) {
133                    StringBuilder bui = new StringBuilder( 100 );
134                    if( handshakedata instanceof ClientHandshake ) {
135                            bui.append( "GET " );
136                            bui.append( ( (ClientHandshake) handshakedata ).getResourceDescriptor() );
137                            bui.append( " HTTP/1.1" );
138                    } else if( handshakedata instanceof ServerHandshake ) {
139                            bui.append( "HTTP/1.1 101 " + ( (ServerHandshake) handshakedata ).getHttpStatusMessage() );
140                    } else {
141                            throw new RuntimeException( "unknow role" );
142                    }
143                    bui.append( "\r\n" );
144                    Iterator<String> it = handshakedata.iterateHttpFields();
145                    while ( it.hasNext() ) {
146                            String fieldname = it.next();
147                            String fieldvalue = handshakedata.getFieldValue( fieldname );
148                            bui.append( fieldname );
149                            bui.append( ": " );
150                            bui.append( fieldvalue );
151                            bui.append( "\r\n" );
152                    }
153                    bui.append( "\r\n" );
154                    byte[] httpheader = Charsetfunctions.asciiBytes( bui.toString() );
155    
156                    byte[] content = withcontent ? handshakedata.getContent() : null;
157                    ByteBuffer bytebuffer = ByteBuffer.allocate( ( content == null ? 0 : content.length ) + httpheader.length );
158                    bytebuffer.put( httpheader );
159                    if( content != null )
160                            bytebuffer.put( content );
161                    bytebuffer.flip();
162                    return Collections.singletonList( bytebuffer );
163            }
