028    public class Draft_76 extends Draft_75 {
029            private boolean failed = false;
030            private static final byte[] closehandshake = { -1, 0 };

032            private final Random reuseableRandom = new Random();

035            public static byte[] createChallenge( String key1, String key2, byte[] key3 ) throws InvalidHandshakeException {
036                    byte[] part1 = getPart( key1 );
037                    byte[] part2 = getPart( key2 );
038                    byte[] challenge = new byte[ 16 ];
039                    challenge[ 0 ] = part1[ 0 ];
040                    challenge[ 1 ] = part1[ 1 ];
041                    challenge[ 2 ] = part1[ 2 ];
042                    challenge[ 3 ] = part1[ 3 ];
043                    challenge[ 4 ] = part2[ 0 ];
044                    challenge[ 5 ] = part2[ 1 ];
045                    challenge[ 6 ] = part2[ 2 ];
046                    challenge[ 7 ] = part2[ 3 ];
047                    challenge[ 8 ] = key3[ 0 ];
048                    challenge[ 9 ] = key3[ 1 ];
049                    challenge[ 10 ] = key3[ 2 ];
050                    challenge[ 11 ] = key3[ 3 ];
051                    challenge[ 12 ] = key3[ 4 ];
052                    challenge[ 13 ] = key3[ 5 ];
053                    challenge[ 14 ] = key3[ 6 ];
054                    challenge[ 15 ] = key3[ 7 ];
055                    MessageDigest md5;
056                    try {
057                            md5 = MessageDigest.getInstance( "MD5" );
058                    } catch ( NoSuchAlgorithmException e ) {
059                            throw new RuntimeException( e );
060                    }
061                    return md5.digest( challenge );
062            }
063    
064            private static String generateKey() {
065                    Random r = new Random();
066                    long maxNumber = 4294967295L;
067                    long spaces = r.nextInt( 12 ) + 1;
068                    int max = new Long( maxNumber / spaces ).intValue();
069                    max = Math.abs( max );
070                    int number = r.nextInt( max ) + 1;
071                    long product = number * spaces;
072                    String key = Long.toString( product );
073                    // always insert atleast one random character
074                    int numChars = r.nextInt( 12 ) + 1;
075                    for( int i = 0 ; i < numChars ; i++ ) {
076                            int position = r.nextInt( key.length() );
077                            position = Math.abs( position );
078                            char randChar = (char) ( r.nextInt( 95 ) + 33 );
079                            // exclude numbers here
080                            if( randChar >= 48 && randChar <= 57 ) {
081                                    randChar -= 15;
082                            }
083                            key = new StringBuilder( key ).insert( position, randChar ).toString();
084                    }
085                    for( int i = 0 ; i < spaces ; i++ ) {
086                            int position = r.nextInt( key.length() - 1 ) + 1;
087                            position = Math.abs( position );
088                            key = new StringBuilder( key ).insert( position, "\u0020" ).toString();
089                    }
090                    return key;
091            }
092    
093            private static byte[] getPart( String key ) throws InvalidHandshakeException {
094                    try {
095                            long keyNumber = Long.parseLong( key.replaceAll( "[^0-9]", "" ) );
096                            long keySpace = key.split( "\u0020" ).length - 1;
097                            if( keySpace == 0 ) {
098                                    throw new InvalidHandshakeException( "invalid Sec-WebSocket-Key (/key2/)" );
099                            }
100                            long part = new Long( keyNumber / keySpace );
101                            return new byte[]{ (byte) ( part >> 24 ), (byte) ( ( part << 8 ) >> 24 ), (byte) ( ( part << 16 ) >> 24 ), (byte) ( ( part << 24 ) >> 24 ) };
102                    } catch ( NumberFormatException e ) {
103                            throw new InvalidHandshakeException( "invalid Sec-WebSocket-Key (/key1/ or /key2/)" );
104                    }
105            }

107            @Override
108            public HandshakeState acceptHandshakeAsClient( ClientHandshake request, ServerHandshake response ) {
109                    if( failed ) {
110                            return HandshakeState.NOT_MATCHED;
111                    }
112    
113                    try {
114                            if( !response.getFieldValue( "Sec-WebSocket-Origin" ).equals( request.getFieldValue( "Origin" ) ) || !basicAccept( response ) ) {
115                                    return HandshakeState.NOT_MATCHED;
116                            }
117                            byte[] content = response.getContent();
118                            if( content == null || content.length == 0 ) {
119                                    throw new IncompleteHandshakeException();
120                            }
121                            if( Arrays.equals( content, createChallenge( request.getFieldValue( "Sec-WebSocket-Key1" ), request.getFieldValue( "Sec-WebSocket-Key2" ), request.getContent() ) ) ) {
122                                    return HandshakeState.MATCHED;
123                            } else {
124                                    return HandshakeState.NOT_MATCHED;
125                            }
126                    } catch ( InvalidHandshakeException e ) {
127                            throw new RuntimeException( "bad handshakerequest", e );
128                    }
129            }
130    
131            @Override
132            public HandshakeState acceptHandshakeAsServer( ClientHandshake handshakedata ) {
133                    if( handshakedata.getFieldValue( "Upgrade" ).equals( "WebSocket" ) && handshakedata.getFieldValue( "Connection" ).contains( "Upgrade" ) && handshakedata.getFieldValue( "Sec-WebSocket-Key1" ).length() > 0 && !handshakedata.getFieldValue( "Sec-WebSocket-Key2" ).isEmpty() && handshakedata.hasFieldValue( "Origin" ) )
134                            return HandshakeState.MATCHED;
135                    return HandshakeState.NOT_MATCHED;
136            }

138            @Override
139            public ClientHandshakeBuilder postProcessHandshakeRequestAsClient( ClientHandshakeBuilder request ) {
140                    request.put( "Upgrade", "WebSocket" );
141                    request.put( "Connection", "Upgrade" );
142                    request.put( "Sec-WebSocket-Key1", generateKey() );
143                    request.put( "Sec-WebSocket-Key2", generateKey() );
144    
145                    if( !request.hasFieldValue( "Origin" ) ) {
146                            request.put( "Origin", "random" + reuseableRandom.nextInt() );
147                    }
148    
149                    byte[] key3 = new byte[ 8 ];
150                    reuseableRandom.nextBytes( key3 );
151                    request.setContent( key3 );
152                    return request;
153    
154            }
155    
156            @Override
157            public HandshakeBuilder postProcessHandshakeResponseAsServer( ClientHandshake request, ServerHandshakeBuilder response ) throws InvalidHandshakeException {
158                    response.setHttpStatusMessage( "WebSocket Protocol Handshake" );
159                    response.put( "Upgrade", "WebSocket" );
160                    response.put( "Connection", request.getFieldValue( "Connection" ) ); // to respond to a Connection keep alive
161                    response.put( "Sec-WebSocket-Origin", request.getFieldValue( "Origin" ) );
162                    String location = "ws://" + request.getFieldValue( "Host" ) + request.getResourceDescriptor();
163                    response.put( "Sec-WebSocket-Location", location );
164                    String key1 = request.getFieldValue( "Sec-WebSocket-Key1" );
165                    String key2 = request.getFieldValue( "Sec-WebSocket-Key2" );
166                    byte[] key3 = request.getContent();
167                    if( key1 == null || key2 == null || key3 == null || key3.length != 8 ) {
168                            throw new InvalidHandshakeException( "Bad keys" );
169                    }
170                    response.setContent( createChallenge( key1, key2, key3 ) );
171                    return response;
172            }

174            @Override
175            public Handshakedata translateHandshake( ByteBuffer buf ) throws InvalidHandshakeException {
176    
177                    HandshakeBuilder bui = translateHandshakeHttp( buf, role );
178                    // the first drafts are lacking a protocol number which makes them difficult to distinguish. Sec-WebSocket-Key1 is typical for draft76
179                    if( ( bui.hasFieldValue( "Sec-WebSocket-Key1" ) || role == Role.CLIENT ) && !bui.hasFieldValue( "Sec-WebSocket-Version" ) ) {
180                            byte[] key3 = new byte[ role == Role.SERVER ? 8 : 16 ];
181                            try {
182                                    buf.get( key3 );
183                            } catch ( BufferUnderflowException e ) {
184                                    throw new IncompleteHandshakeException( buf.capacity() + 16 );
185                            }
186                            bui.setContent( key3 );
187    
188                    }
189                    return bui;
190            }

192            @Override
193            public List<Framedata> translateFrame( ByteBuffer buffer ) throws InvalidDataException {
194                    buffer.mark();
195                    List<Framedata> frames = super.translateRegularFrame( buffer );
196                    if( frames == null ) {
197                            buffer.reset();
198                            frames = readyframes;
199                            readingState = true;
200                            if( currentFrame == null )
201                                    currentFrame = ByteBuffer.allocate( 2 );
202                            else {
203                                    throw new InvalidFrameException();
204                            }
205                            if( buffer.remaining() > currentFrame.remaining() ) {
206                                    throw new InvalidFrameException();
207                            } else {
208                                    currentFrame.put( buffer );
209                            }
210                            if( !currentFrame.hasRemaining() ) {
211                                    if( Arrays.equals( currentFrame.array(), closehandshake ) ) {
212                                            frames.add( new CloseFrameBuilder( CloseFrame.NORMAL ) );
213                                            return frames;
214                                    }
215                                    else{
216                                            throw new InvalidFrameException();
217                                    }
218                            } else {
219                                    readyframes = new LinkedList<Framedata>();
220                                    return frames;
221                            }
222                    } else {
223                            return frames;
224                    }
225            }
226            @Override
227            public ByteBuffer createBinaryFrame( Framedata framedata ) {
228                    if( framedata.getOpcode() == Opcode.CLOSING )
229                            return ByteBuffer.wrap( closehandshake );
230                    return super.createBinaryFrame( framedata );
231            }
