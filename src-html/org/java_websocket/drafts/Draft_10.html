050            public static int readVersion( Handshakedata handshakedata ) {
051                    String vers = handshakedata.getFieldValue( "Sec-WebSocket-Version" );
052                    if( vers.length() > 0 ) {
053                            int v;
054                            try {
055                                    v = new Integer( vers.trim() );
056                                    return v;
057                            } catch ( NumberFormatException e ) {
058                                    return -1;
059                            }
060                    }
061                    return -1;
062            }

069            @Override
070            public HandshakeState acceptHandshakeAsClient( ClientHandshake request, ServerHandshake response ) throws InvalidHandshakeException {
071                    if( !request.hasFieldValue( "Sec-WebSocket-Key" ) || !response.hasFieldValue( "Sec-WebSocket-Accept" ) )
072                            return HandshakeState.NOT_MATCHED;
073    
074                    String seckey_answere = response.getFieldValue( "Sec-WebSocket-Accept" );
075                    String seckey_challenge = request.getFieldValue( "Sec-WebSocket-Key" );
076                    seckey_challenge = generateFinalKey( seckey_challenge );
077    
078                    if( seckey_challenge.equals( seckey_answere ) )
079                            return HandshakeState.MATCHED;
080                    return HandshakeState.NOT_MATCHED;
081            }

085                    // Sec-WebSocket-Origin is only required for browser clients

087                    if( v == 7 || v == 8 )// g
088                            return basicAccept( handshakedata ) ? HandshakeState.MATCHED : HandshakeState.NOT_MATCHED;
089                    return HandshakeState.NOT_MATCHED;

092            @Override
093            public ByteBuffer createBinaryFrame( Framedata framedata ) {
094                    ByteBuffer mes = framedata.getPayloadData();
095                    boolean mask = role == Role.CLIENT; // framedata.getTransfereMasked();
096                    int sizebytes = mes.remaining() <= 125 ? 1 : mes.remaining() <= 65535 ? 2 : 8;
097                    ByteBuffer buf = ByteBuffer.allocate( 1 + ( sizebytes > 1 ? sizebytes + 1 : sizebytes ) + ( mask ? 4 : 0 ) + mes.remaining() );
098                    byte optcode = fromOpcode( framedata.getOpcode() );
099                    byte one = (byte) ( framedata.isFin() ? -128 : 0 );
100                    one |= optcode;
101                    buf.put( one );
102                    byte[] payloadlengthbytes = toByteArray( mes.remaining(), sizebytes );
103                    assert ( payloadlengthbytes.length == sizebytes );
104    
105                    if( sizebytes == 1 ) {
106                            buf.put( (byte) ( (byte) payloadlengthbytes[ 0 ] | ( mask ? (byte) -128 : 0 ) ) );
107                    } else if( sizebytes == 2 ) {
108                            buf.put( (byte) ( (byte) 126 | ( mask ? (byte) -128 : 0 ) ) );
109                            buf.put( payloadlengthbytes );
110                    } else if( sizebytes == 8 ) {
111                            buf.put( (byte) ( (byte) 127 | ( mask ? (byte) -128 : 0 ) ) );
112                            buf.put( payloadlengthbytes );
113                    } else
114                            throw new RuntimeException( "Size representation not supported/specified" );
115    
116                    if( mask ) {
117                            ByteBuffer maskkey = ByteBuffer.allocate( 4 );
118                            maskkey.putInt( reuseableRandom.nextInt() );
119                            buf.put( maskkey.array() );
120                            for( int i = 0 ; i < mes.limit() ; i++ ) {
121                                    buf.put( (byte) ( mes.get() ^ maskkey.get( i % 4 ) ) );
122                            }
123                    } else
124                            buf.put( mes );
125                    // translateFrame ( buf.array () , buf.array ().length );
126                    assert ( buf.remaining() == 0 ) : buf.remaining();
127                    buf.flip();
128    
129                    return buf;
130            }

160            private byte fromOpcode( Opcode opcode ) {
161                    if( opcode == Opcode.CONTINUOUS )
162                            return 0;
163                    else if( opcode == Opcode.TEXT )
164                            return 1;
165                    else if( opcode == Opcode.BINARY )
166                            return 2;
167                    else if( opcode == Opcode.CLOSING )
168                            return 8;
169                    else if( opcode == Opcode.PING )
170                            return 9;
171                    else if( opcode == Opcode.PONG )
172                            return 10;
173                    throw new RuntimeException( "Don't know how to handle " + opcode.toString() );
174            }

176            private String generateFinalKey( String in ) {
177                    String seckey = in.trim();
178                    String acc = seckey + "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
179                    MessageDigest sh1;
180                    try {
181                            sh1 = MessageDigest.getInstance( "SHA1" );
182                    } catch ( NoSuchAlgorithmException e ) {
183                            throw new RuntimeException( e );
184                    }
185                    return Base64.encodeBytes( sh1.digest( acc.getBytes() ) );
186            }

202            public HandshakeBuilder postProcessHandshakeResponseAsServer( ClientHandshake request, ServerHandshakeBuilder response ) throws InvalidHandshakeException {
203                    response.put( "Upgrade", "websocket" );
204                    response.put( "Connection", request.getFieldValue( "Connection" ) ); // to respond to a Connection keep alives
205                    response.setHttpStatusMessage( "Switching Protocols" );
206                    String seckey = request.getFieldValue( "Sec-WebSocket-Key" );
207                    if( seckey == null )
208                            throw new InvalidHandshakeException( "missing Sec-WebSocket-Key" );
209                    response.put( "Sec-WebSocket-Accept", generateFinalKey( seckey ) );
210                    return response;
211            }

213            private byte[] toByteArray( long val, int bytecount ) {
214                    byte[] buffer = new byte[ bytecount ];
215                    int highest = 8 * bytecount - 8;
216                    for( int i = 0 ; i < bytecount ; i++ ) {
217                            buffer[ i ] = (byte) ( val >>> ( highest - 8 * i ) );
218                    }
219                    return buffer;
220            }

222            private Opcode toOpcode( byte opcode ) throws InvalidFrameException {
223                    switch ( opcode ) {
224                            case 0:
225                                    return Opcode.CONTINUOUS;
226                            case 1:
227                                    return Opcode.TEXT;
228                            case 2:
229                                    return Opcode.BINARY;
230                                    // 3-7 are not yet defined
231                            case 8:
232                                    return Opcode.CLOSING;
233                            case 9:
234                                    return Opcode.PING;
235                            case 10:
236                                    return Opcode.PONG;
237                                    // 11-15 are not yet defined
238                            default :
239                                    throw new InvalidFrameException( "unknow optcode " + (short) opcode );
240                    }
241            }

243            @Override
244            public List<Framedata> translateFrame( ByteBuffer buffer ) throws LimitExedeedException , InvalidDataException {
245                    List<Framedata> frames = new LinkedList<Framedata>();
246                    Framedata cur;
247    
248                    if( incompleteframe != null ) {
249                            // complete an incomplete frame
250                            while ( true ) {
251                                    try {
252                                            buffer.mark();
253                                            int available_next_byte_count = buffer.remaining();// The number of bytes received
254                                            int expected_next_byte_count = incompleteframe.remaining();// The number of bytes to complete the incomplete frame
255    
256                                            if( expected_next_byte_count > available_next_byte_count ) {
257                                                    // did not receive enough bytes to complete the frame
258                                                    incompleteframe.put( buffer.array(), buffer.position(), available_next_byte_count );
259                                                    buffer.position( buffer.position() + available_next_byte_count );
260                                                    return Collections.emptyList();
261                                            }
262                                            incompleteframe.put( buffer.array(), buffer.position(), expected_next_byte_count );
263                                            buffer.position( buffer.position() + expected_next_byte_count );
264    
265                                            cur = translateSingleFrame( (ByteBuffer) incompleteframe.duplicate().position( 0 ) );
266                                            frames.add( cur );
267                                            incompleteframe = null;
268                                            break; // go on with the normal frame receival
269                                    } catch ( IncompleteException e ) {
270                                            // extending as much as suggested
271                                            int oldsize = incompleteframe.limit();
272                                            ByteBuffer extendedframe = ByteBuffer.allocate( checkAlloc( e.getPreferedSize() ) );
273                                            assert ( extendedframe.limit() > incompleteframe.limit() );
274                                            incompleteframe.rewind();
275                                            extendedframe.put( incompleteframe );
276                                            incompleteframe = extendedframe;
277                                            
278                                            return translateFrame( buffer );
279                                    }
280                            }
281                    }
282    
283                    while ( buffer.hasRemaining() ) {// Read as much as possible full frames
284                            buffer.mark();
285                            try {
286                                    cur = translateSingleFrame( buffer );
287                                    frames.add( cur );
288                            } catch ( IncompleteException e ) {
289                                    // remember the incomplete data
290                                    buffer.reset();
291                                    int pref = e.getPreferedSize();
292                                    incompleteframe = ByteBuffer.allocate( checkAlloc( pref ) );
293                                    incompleteframe.put( buffer );
294                                    break;
295                            }
296                    }
297                    return frames;
298            }

300            public Framedata translateSingleFrame( ByteBuffer buffer ) throws IncompleteException , InvalidDataException {
301                    int maxpacketsize = buffer.remaining();
302                    int realpacketsize = 2;
303                    if( maxpacketsize < realpacketsize )
304                            throw new IncompleteException( realpacketsize );
305                    byte b1 = buffer.get( /*0*/);
306                    boolean FIN = b1 >> 8 != 0;
307                    byte rsv = (byte) ( ( b1 & ~(byte) 128 ) >> 4 );
308                    if( rsv != 0 )
309                            throw new InvalidFrameException( "bad rsv " + rsv );
310                    byte b2 = buffer.get( /*1*/);
311                    boolean MASK = ( b2 & -128 ) != 0;
312                    int payloadlength = (byte) ( b2 & ~(byte) 128 );
313                    Opcode optcode = toOpcode( (byte) ( b1 & 15 ) );
314    
315                    if( !FIN ) {
316                            if( optcode == Opcode.PING || optcode == Opcode.PONG || optcode == Opcode.CLOSING ) {
317                                    throw new InvalidFrameException( "control frames may no be fragmented" );
318                            }
319                    }
320    
321                    if( payloadlength >= 0 && payloadlength <= 125 ) {
322                    } else {
323                            if( optcode == Opcode.PING || optcode == Opcode.PONG || optcode == Opcode.CLOSING ) {
324                                    throw new InvalidFrameException( "more than 125 octets" );
325                            }
326                            if( payloadlength == 126 ) {
327                                    realpacketsize += 2; // additional length bytes
328                                    if( maxpacketsize < realpacketsize )
329                                            throw new IncompleteException( realpacketsize );
330                                    byte[] sizebytes = new byte[ 3 ];
331                                    sizebytes[ 1 ] = buffer.get( /*1 + 1*/);
332                                    sizebytes[ 2 ] = buffer.get( /*1 + 2*/);
333                                    payloadlength = new BigInteger( sizebytes ).intValue();
334                            } else {
335                                    realpacketsize += 8; // additional length bytes
336                                    if( maxpacketsize < realpacketsize )
337                                            throw new IncompleteException( realpacketsize );
338                                    byte[] bytes = new byte[ 8 ];
339                                    for( int i = 0 ; i < 8 ; i++ ) {
340                                            bytes[ i ] = buffer.get( /*1 + i*/);
341                                    }
342                                    long length = new BigInteger( bytes ).longValue();
343                                    if( length > Integer.MAX_VALUE ) {
344                                            throw new LimitExedeedException( "Payloadsize is to big..." );
345                                    } else {
346                                            payloadlength = (int) length;
347                                    }
348                            }
349                    }
350    
351                    // int maskskeystart = foff + realpacketsize;
352                    realpacketsize += ( MASK ? 4 : 0 );
353                    // int payloadstart = foff + realpacketsize;
354                    realpacketsize += payloadlength;
355    
356                    if( maxpacketsize < realpacketsize )
357                            throw new IncompleteException( realpacketsize );
358    
359                    ByteBuffer payload = ByteBuffer.allocate( checkAlloc( payloadlength ) );
360                    if( MASK ) {
361                            byte[] maskskey = new byte[ 4 ];
362                            buffer.get( maskskey );
363                            for( int i = 0 ; i < payloadlength ; i++ ) {
364                                    payload.put( (byte) ( (byte) buffer.get( /*payloadstart + i*/) ^ (byte) maskskey[ i % 4 ] ) );
365                            }
366                    } else {
367                            payload.put( buffer.array(), buffer.position(), payload.limit() );
368                            buffer.position( buffer.position() + payload.limit() );
369                    }
370    
371                    FrameBuilder frame;
372                    if( optcode == Opcode.CLOSING ) {
373                            frame = new CloseFrameBuilder();
374                    } else {
375                            frame = new FramedataImpl1();
376                            frame.setFin( FIN );
377                            frame.setOptcode( optcode );
378                    }
379                    payload.flip();
380                    frame.setPayload( payload );
381                    return frame;
382            }
