024    public class Draft_75 extends Draft {
025    
026            /**
027             * The byte representing CR, or Carriage Return, or \r
028             */
029            public static final byte CR = (byte) 0x0D;
030            /**
031             * The byte representing LF, or Line Feed, or \n
032             */
033            public static final byte LF = (byte) 0x0A;
034            /**
035             * The byte representing the beginning of a WebSocket text frame.
036             */
037            public static final byte START_OF_FRAME = (byte) 0x00;
038            /**
039             * The byte representing the end of a WebSocket text frame.
040             */
041            public static final byte END_OF_FRAME = (byte) 0xFF;
042    
043            protected boolean readingState = false;
044            private boolean inframe = false;
045            protected List<Framedata> readyframes = new LinkedList<Framedata>();
046            protected ByteBuffer currentFrame;
047            
048            
049            private final Random reuseableRandom = new Random();
050

052            @Override
053            public HandshakeState acceptHandshakeAsClient( ClientHandshake request, ServerHandshake response ) {
054                    return request.getFieldValue( "WebSocket-Origin" ).equals( response.getFieldValue( "Origin" ) ) && basicAccept( response ) ? HandshakeState.MATCHED : HandshakeState.NOT_MATCHED;
055            }
056    
057            @Override
058            public HandshakeState acceptHandshakeAsServer( ClientHandshake handshakedata ) {
059                    if( handshakedata.hasFieldValue( "Origin" ) && basicAccept( handshakedata ) ) {
060                            return HandshakeState.MATCHED;
061                    }
062                    return HandshakeState.NOT_MATCHED;
063            }
064    
065            @Override
066            public ByteBuffer createBinaryFrame( Framedata framedata ) {
067                    if( framedata.getOpcode() != Opcode.TEXT ) {
068                            throw new RuntimeException( "only text frames supported" );
069                    }
070    
071                    ByteBuffer pay = framedata.getPayloadData();
072                    ByteBuffer b = ByteBuffer.allocate( pay.remaining() + 2 );
073                    b.put( START_OF_FRAME );
074                    pay.mark();
075                    b.put( pay );
076                    pay.reset();
077                    b.put( END_OF_FRAME );
078                    b.flip();
079                    return b;
080            }
081    
082            @Override
083            public List<Framedata> createFrames( ByteBuffer binary, boolean mask ) {
084                    throw new RuntimeException( "not yet implemented" );
085            }
086    
087            @Override
088            public List<Framedata> createFrames( String text, boolean mask ) {
089                    FrameBuilder frame = new FramedataImpl1();
090                    try {
091                            frame.setPayload( ByteBuffer.wrap( Charsetfunctions.utf8Bytes( text ) ) );
092                    } catch ( InvalidDataException e ) {
093                            throw new NotSendableException( e );
094                    }
095                    frame.setFin( true );
096                    frame.setOptcode( Opcode.TEXT );
097                    frame.setTransferemasked( mask );
098                    return Collections.singletonList( (Framedata) frame );
099            }
100    
101            @Override
102            public ClientHandshakeBuilder postProcessHandshakeRequestAsClient( ClientHandshakeBuilder request ) throws InvalidHandshakeException {
103                    request.put( "Upgrade", "WebSocket" );
104                    request.put( "Connection", "Upgrade" );
105                    if( !request.hasFieldValue( "Origin" ) ) {
106                            request.put( "Origin", "random" + reuseableRandom.nextInt() );
107                    }
108    
109                    return request;
110            }
111    
112            @Override
113            public HandshakeBuilder postProcessHandshakeResponseAsServer( ClientHandshake request, ServerHandshakeBuilder response ) throws InvalidHandshakeException {
114                    response.setHttpStatusMessage( "Web Socket Protocol Handshake" );
115                    response.put( "Upgrade", "WebSocket" );
116                    response.put( "Connection", request.getFieldValue( "Connection" ) ); // to respond to a Connection keep alive
117                    response.put( "WebSocket-Origin", request.getFieldValue( "Origin" ) );
118                    String location = "ws://" + request.getFieldValue( "Host" ) + request.getResourceDescriptor();
119                    response.put( "WebSocket-Location", location );
120                    // TODO handle Sec-WebSocket-Protocol and Set-Cookie
121                    return response;
122            }
123    
124            protected List<Framedata> translateRegularFrame( ByteBuffer buffer ) throws InvalidDataException {
125                    while ( buffer.hasRemaining() ) {
126                            byte newestByte = buffer.get();
127                            if( newestByte == START_OF_FRAME ) { // Beginning of Frame
128                                    if( readingState )
129                                            return null;
130                                    readingState = true;
131                            } else if( newestByte == END_OF_FRAME ) { // End of Frame
132                                    if( !readingState )
133                                            return null;
134                                    // currentFrame will be null if END_OF_FRAME was send directly after
135                                    // START_OF_FRAME, thus we will send 'null' as the sent message.
136                                    if( this.currentFrame != null ) {
137                                            currentFrame.flip();
138                                            FramedataImpl1 curframe = new FramedataImpl1();
139                                            curframe.setPayload( currentFrame );
140                                            curframe.setFin( true );
141                                            curframe.setOptcode( inframe ? Opcode.CONTINUOUS : Opcode.TEXT );
142                                            readyframes.add( curframe );
143                                            this.currentFrame = null;
144                                            buffer.mark();
145                                    }
146                                    readingState = false;
147                                    inframe = false;
148                            } else if( readingState ) { // Regular frame data, add to current frame buffer //TODO This code is very expensive and slow
149                                    if( currentFrame == null ) {
150                                            currentFrame = createBuffer();
151                                    } else if( !currentFrame.hasRemaining() ) {
152                                            currentFrame = increaseBuffer( currentFrame );
153                                    }
154                                    currentFrame.put( newestByte );
155                            } else {
156                                    return null;
157                            }
158                    }
159                    if( readingState ) {
160                            FramedataImpl1 curframe = new FramedataImpl1();
161                            currentFrame.flip();
162                            curframe.setPayload( currentFrame );
163                            curframe.setFin( false );
164                            curframe.setOptcode( inframe ? Opcode.CONTINUOUS : Opcode.TEXT );
165                            inframe = true;
166                            readyframes.add( curframe );
167                    }
168    
169                    List<Framedata> frames = readyframes;
170                    readyframes = new LinkedList<Framedata>();
171                    this.currentFrame = null;
172                    return frames;
173            }
174    
175            @Override
176            public List<Framedata> translateFrame( ByteBuffer buffer ) throws InvalidDataException {
177                    List<Framedata> frames = translateRegularFrame( buffer );
178                    if( frames == null ) {
179                            throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR );
180                    }
181                    return frames;
182            }
183    
184            @Override
185            public void reset() {
186                    readingState = false;
187                    this.currentFrame = null;
188            }
189    
190            @Override
191            public CloseHandshakeType getCloseHandshakeType() {
192                    return CloseHandshakeType.NONE;
193            }
194    
195            public ByteBuffer createBuffer() {
196                    return ByteBuffer.allocate( INITIAL_FAMESIZE );
197            }
198    
199            public ByteBuffer increaseBuffer( ByteBuffer full ) {
200                    full.flip();
201                    ByteBuffer newbuffer = ByteBuffer.allocate( full.capacity() * 2 );
202                    newbuffer.put( full );
203                    return newbuffer;
204            }
205    
206            @Override
207            public Draft copyInstance() {
208                    return new Draft_75();
209            }
