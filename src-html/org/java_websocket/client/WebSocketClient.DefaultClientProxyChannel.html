032    /**
033     * The <tt>WebSocketClient</tt> is an abstract class that expects a valid
034     * "ws://" URI to connect to. When connected, an instance recieves important
035     * events related to the life of the connection. A subclass must implement
036     * <var>onOpen</var>, <var>onClose</var>, and <var>onMessage</var> to be
037     * useful. An instance can send messages to it's connected server via the
038     * <var>send</var> method.
039     * 
040     * @author Nathan Rajlich
041     */
042    public abstract class WebSocketClient extends WebSocketAdapter implements Runnable {

088            public WebSocketClient( URI serverUri , Draft draft , Map<String,String> headers , int connecttimeout ) {
089                    if( serverUri == null ) {
090                            throw new IllegalArgumentException();
091                    }
092                    if( draft == null ) {
093                            throw new IllegalArgumentException( "null as draft is permitted for `WebSocketServer` only!" );
094                    }
095                    this.uri = serverUri;
096                    this.draft = draft;
097                    this.headers = headers;
098                    this.timeout = connecttimeout;
099    
100                    try {
101                            channel = SelectorProvider.provider().openSocketChannel();
102                            channel.configureBlocking( true );
103                    } catch ( IOException e ) {
104                            channel = null;
105                            onWebsocketError( null, e );
106                    }
107                    if(channel == null){
108                            conn = (WebSocketImpl) wsfactory.createWebSocket( this, draft, null );
109                            conn.close( CloseFrame.NEVER_CONNECTED, "Failed to create or configure SocketChannel." );
110                    }
111                    else{
112                            conn = (WebSocketImpl) wsfactory.createWebSocket( this, draft, channel.socket() );
113                    }
114                    
115            }

137                    if( writethread != null )

139                    writethread = new Thread( this );

184            // Runnable IMPLEMENTATION /////////////////////////////////////////////////
185            public void run() {
186                    if( writethread == null )
187                            writethread = Thread.currentThread();
188                    interruptableRun();
189    
190                    assert ( !channel.isOpen() );
191    
192            }

194            private final void interruptableRun() {
195                    if( channel == null ) {
196                            return;// channel will be initialized in the constructor and only be null if no socket channel could be created or if blocking mode could be established
197                    }
198    
199                    try {
200                            String host;
201                            int port ;
202    
203                            if( proxyAddress != null ) {
204                                    host = proxyAddress.getHostName();
205                                    port = proxyAddress.getPort();
206                            } else {
207                                    host = uri.getHost();
208                                    port = getPort();
209                            }
210                            channel.connect( new InetSocketAddress( host, port ) );
211                            conn.channel = wrappedchannel = createProxyChannel( wsfactory.wrapChannel( channel, null, host, port ) );
212    
213                            timeout = 0; // since connect is over
214                            sendHandshake();
215                            readthread = new Thread( new WebsocketWriteThread() );
216                            readthread.start();
217                    } catch ( ClosedByInterruptException e ) {
218                            onWebsocketError( null, e );
219                            return;
220                    } catch ( /*IOException | SecurityException | UnresolvedAddressException*/Exception e ) {//
221                            onWebsocketError( conn, e );
222                            conn.closeConnection( CloseFrame.NEVER_CONNECTED, e.getMessage() );
223                            return;
224                    }
225    
226                    ByteBuffer buff = ByteBuffer.allocate( WebSocketImpl.RCVBUF );
227                    try/*IO*/{
228                            while ( channel.isOpen() ) {
229                                    if( SocketChannelIOHelper.read( buff, this.conn, wrappedchannel ) ) {
230                                            conn.decode( buff );
231                                    } else {
232                                            conn.eot();
233                                    }
234    
235                                    if( wrappedchannel instanceof WrappedByteChannel ) {
236                                            WrappedByteChannel w = (WrappedByteChannel) wrappedchannel;
237                                            if( w.isNeedRead() ) {
238                                                    while ( SocketChannelIOHelper.readMore( buff, conn, w ) ) {
239                                                            conn.decode( buff );
240                                                    }
241                                                    conn.decode( buff );
242                                            }
243                                    }
244                            }
245    
246                    } catch ( CancelledKeyException e ) {
247                            conn.eot();
248                    } catch ( IOException e ) {
249                            conn.eot();
250                    } catch ( RuntimeException e ) {
251                            // this catch case covers internal errors only and indicates a bug in this websocket implementation
252                            onError( e );
253                            conn.closeConnection( CloseFrame.ABNORMAL_CLOSE, e.getMessage() );
254                    }
255            }

257            private int getPort() {
258                    int port = uri.getPort();
259                    if( port == -1 ) {
260                            String scheme = uri.getScheme();
261                            if( scheme.equals( "wss" ) ) {
262                                    return WebSocket.DEFAULT_WSS_PORT;
263                            } else if( scheme.equals( "ws" ) ) {
264                                    return WebSocket.DEFAULT_PORT;
265                            } else {
266                                    throw new RuntimeException( "unkonow scheme" + scheme );
267                            }
268                    }
269                    return port;
270            }
271    
272            private void sendHandshake() throws InvalidHandshakeException {
273                    String path;
274                    String part1 = uri.getPath();
275                    String part2 = uri.getQuery();
276                    if( part1 == null || part1.length() == 0 )
277                            path = "/";
278                    else
279                            path = part1;
280                    if( part2 != null )
281                            path += "?" + part2;
282                    int port = getPort();
283                    String host = uri.getHost() + ( port != WebSocket.DEFAULT_PORT ? ":" + port : "" );
284    
285                    HandshakeImpl1Client handshake = new HandshakeImpl1Client();
286                    handshake.setResourceDescriptor( path );
287                    handshake.put( "Host", host );
288                    if( headers != null ) {
289                            for( Map.Entry<String,String> kv : headers.entrySet() ) {
290                                    handshake.put( kv.getKey(), kv.getValue() );
291                            }
292                    }
293                    conn.startHandshake( handshake );
294            }

341                            readthread.interrupt();
342                    onClose( code, reason, remote );

410            public class DefaultClientProxyChannel extends AbstractClientProxyChannel {
411                    public DefaultClientProxyChannel( ByteChannel towrap ) {
412                            super( towrap );
413                    }
414                    @Override
415                    public String buildHandShake() {
416                            StringBuilder b = new StringBuilder();
417                            String host = uri.getHost();
418                            b.append( "CONNECT " );
419                            b.append( host );
420                            b.append( ":" );
421                            b.append( getPort() );
422                            b.append( " HTTP/1.1\n" );
423                            b.append( "Host: " );
424                            b.append( host );
425                            b.append( "\n" );
426                            return b.toString();
427                    }
428            }

434            private class WebsocketWriteThread implements Runnable {
435                    @Override
436                    public void run() {
437                            Thread.currentThread().setName( "WebsocketWriteThread" );
438                            try {
439                                    while ( !Thread.interrupted() ) {
440                                            SocketChannelIOHelper.writeBlocking( conn, wrappedchannel );
441                                    }
442                            } catch ( IOException e ) {
443                                    conn.eot();
444                            } catch ( InterruptedException e ) {
445                                    // this thread is regularly terminated via an interrupt
446                            }
447                    }
448            }

450            public ByteChannel createProxyChannel( ByteChannel towrap ) {
451                    if( proxyAddress != null ){
452                            return new DefaultClientProxyChannel( towrap );
453                    }
454                    return towrap;//no proxy in use
455            }
