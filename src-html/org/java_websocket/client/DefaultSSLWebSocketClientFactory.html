001    package org.java_websocket.client;
002    import java.io.IOException;
003    import java.net.Socket;
004    import java.nio.channels.ByteChannel;
005    import java.nio.channels.SelectionKey;
006    import java.nio.channels.SocketChannel;
007    import java.util.List;
008    import java.util.concurrent.ExecutorService;
009    import java.util.concurrent.Executors;
010    
011    import javax.net.ssl.SSLContext;
012    import javax.net.ssl.SSLEngine;
013    
014    import org.java_websocket.SSLSocketChannel2;
015    import org.java_websocket.WebSocketAdapter;
016    import org.java_websocket.WebSocketImpl;
017    import org.java_websocket.client.WebSocketClient.WebSocketClientFactory;
018    import org.java_websocket.drafts.Draft;
019    
020    
021    public class DefaultSSLWebSocketClientFactory implements WebSocketClientFactory {
022            protected SSLContext sslcontext;
023            protected ExecutorService exec;
024    
025            public DefaultSSLWebSocketClientFactory( SSLContext sslContext ) {
026                    this( sslContext, Executors.newSingleThreadScheduledExecutor() );
027            }
028    
029            public DefaultSSLWebSocketClientFactory( SSLContext sslContext , ExecutorService exec ) {
030                    if( sslContext == null || exec == null )
031                            throw new IllegalArgumentException();
032                    this.sslcontext = sslContext;
033                    this.exec = exec;
034            }
035    
036            @Override
037            public ByteChannel wrapChannel( SocketChannel channel, SelectionKey key, String host, int port ) throws IOException {
038                    SSLEngine e = sslcontext.createSSLEngine( host, port );
039                    e.setUseClientMode( true );
040                    return new SSLSocketChannel2( channel, e, exec, key );
041            }
042    
043            @Override
044            public WebSocketImpl createWebSocket( WebSocketAdapter a, Draft d, Socket c ) {
045                    return new WebSocketImpl( a, d, c );
046            }
047    
048            @Override
049            public WebSocketImpl createWebSocket( WebSocketAdapter a, List<Draft> d, Socket s ) {
050                    return new WebSocketImpl( a, d, s );
051            }
052    }
