166            /**
167             * Returns whether the handshake phase has is completed.
168             * In case of a broken handshake this will be never the case.
169             **/
170            private boolean decodeHandshake( ByteBuffer socketBufferNew ) {
171                    ByteBuffer socketBuffer;
172                    if( tmpHandshakeBytes == null ) {
173                            socketBuffer = socketBufferNew;
174                    } else {
175                            if( tmpHandshakeBytes.remaining() < socketBufferNew.remaining() ) {
176                                    ByteBuffer buf = ByteBuffer.allocate( tmpHandshakeBytes.capacity() + socketBufferNew.remaining() );
177                                    tmpHandshakeBytes.flip();
178                                    buf.put( tmpHandshakeBytes );
179                                    tmpHandshakeBytes = buf;
180                            }
181    
182                            tmpHandshakeBytes.put( socketBufferNew );
183                            tmpHandshakeBytes.flip();
184                            socketBuffer = tmpHandshakeBytes;
185                    }
186                    socketBuffer.mark();
187                    try {
188                            if( draft == null ) {
189                                    HandshakeState isflashedgecase = isFlashEdgeCase( socketBuffer );
190                                    if( isflashedgecase == HandshakeState.MATCHED ) {
191                                            write( ByteBuffer.wrap( Charsetfunctions.utf8Bytes( wsl.getFlashPolicy( this ) ) ) );
192                                            close( CloseFrame.FLASHPOLICY, "" );
193                                            return false;
194                                    }
195                            }
196                            HandshakeState handshakestate = null;
197    
198                            try {
199                                    if( role == Role.SERVER ) {
200                                            if( draft == null ) {
201                                                    for( Draft d : knownDrafts ) {
202                                                            d = d.copyInstance();
203                                                            try {
204                                                                    d.setParseMode( role );
205                                                                    socketBuffer.reset();
206                                                                    Handshakedata tmphandshake = d.translateHandshake( socketBuffer );
207                                                                    if( tmphandshake instanceof ClientHandshake == false ) {
208                                                                            flushAndClose( CloseFrame.PROTOCOL_ERROR, "wrong http function", false );
209                                                                            return false;
210                                                                    }
211                                                                    ClientHandshake handshake = (ClientHandshake) tmphandshake;
212                                                                    handshakestate = d.acceptHandshakeAsServer( handshake );
213                                                                    if( handshakestate == HandshakeState.MATCHED ) {
214                                                                            ServerHandshakeBuilder response;
215                                                                            try {
216                                                                                    response = wsl.onWebsocketHandshakeReceivedAsServer( this, d, handshake );
217                                                                            } catch ( InvalidDataException e ) {
218                                                                                    flushAndClose( e.getCloseCode(), e.getMessage(), false );
219                                                                                    return false;
220                                                                            } catch ( RuntimeException e ) {
221                                                                                    wsl.onWebsocketError( this, e );
222                                                                                    flushAndClose( CloseFrame.NEVER_CONNECTED, e.getMessage(), false );
223                                                                                    return false;
224                                                                            }
225                                                                            write( d.createHandshake( d.postProcessHandshakeResponseAsServer( handshake, response ), role ) );
226                                                                            draft = d;
227                                                                            open( handshake );
228                                                                            return true;
229                                                                    }
230                                                            } catch ( InvalidHandshakeException e ) {
231                                                                    // go on with an other draft
232                                                            }
233                                                    }
234                                                    if( draft == null ) {
235                                                            close( CloseFrame.PROTOCOL_ERROR, "no draft matches" );
236                                                    }
237                                                    return false;
238                                            } else {
239                                                    // special case for multiple step handshakes
240                                                    Handshakedata tmphandshake = draft.translateHandshake( socketBuffer );
241                                                    if( tmphandshake instanceof ClientHandshake == false ) {
242                                                            flushAndClose( CloseFrame.PROTOCOL_ERROR, "wrong http function", false );
243                                                            return false;
244                                                    }
245                                                    ClientHandshake handshake = (ClientHandshake) tmphandshake;
246                                                    handshakestate = draft.acceptHandshakeAsServer( handshake );
247    
248                                                    if( handshakestate == HandshakeState.MATCHED ) {
249                                                            open( handshake );
250                                                            return true;
251                                                    } else {
252                                                            close( CloseFrame.PROTOCOL_ERROR, "the handshake did finaly not match" );
253                                                    }
254                                                    return false;
255                                            }
256                                    } else if( role == Role.CLIENT ) {
257                                            draft.setParseMode( role );
258                                            Handshakedata tmphandshake = draft.translateHandshake( socketBuffer );
259                                            if( tmphandshake instanceof ServerHandshake == false ) {
260                                                    flushAndClose( CloseFrame.PROTOCOL_ERROR, "Wwrong http function", false );
261                                                    return false;
262                                            }
263                                            ServerHandshake handshake = (ServerHandshake) tmphandshake;
264                                            handshakestate = draft.acceptHandshakeAsClient( handshakerequest, handshake );
265                                            if( handshakestate == HandshakeState.MATCHED ) {
266                                                    try {
267                                                            wsl.onWebsocketHandshakeReceivedAsClient( this, handshakerequest, handshake );
268                                                    } catch ( InvalidDataException e ) {
269                                                            flushAndClose( e.getCloseCode(), e.getMessage(), false );
270                                                            return false;
271                                                    } catch ( RuntimeException e ) {
272                                                            wsl.onWebsocketError( this, e );
273                                                            flushAndClose( CloseFrame.NEVER_CONNECTED, e.getMessage(), false );
274                                                            return false;
275                                                    }
276                                                    open( handshake );
277                                                    return true;
278                                            } else {
279                                                    close( CloseFrame.PROTOCOL_ERROR, "draft " + draft + " refuses handshake" );
280                                            }
281                                    }
282                            } catch ( InvalidHandshakeException e ) {
283                                    close( e );
284                            }
285                    } catch ( IncompleteHandshakeException e ) {
286                            if( tmpHandshakeBytes == null ) {
287                                    socketBuffer.reset();
288                                    int newsize = e.getPreferedSize();
289                                    if( newsize == 0 ) {
290                                            newsize = socketBuffer.capacity() + 16;
291                                    } else {
292                                            assert ( e.getPreferedSize() >= socketBuffer.remaining() );
293                                    }
294                                    tmpHandshakeBytes = ByteBuffer.allocate( newsize );
295    
296                                    tmpHandshakeBytes.put( socketBufferNew );
297                                    // tmpHandshakeBytes.flip();
298                            } else {
299                                    tmpHandshakeBytes.position( tmpHandshakeBytes.limit() );
300                                    tmpHandshakeBytes.limit( tmpHandshakeBytes.capacity() );
301                            }
302                    }
303                    return false;
304            }

306            private void decodeFrames( ByteBuffer socketBuffer ) {
307                    if( flushandclosestate )
308                            return;
309    
310                    List<Framedata> frames;
311                    try {
312                            frames = draft.translateFrame( socketBuffer );
313                            for( Framedata f : frames ) {
314                                    if( DEBUG )
315                                            System.out.println( "matched frame: " + f );
316                                    if( flushandclosestate )
317                                            return;
318                                    Opcode curop = f.getOpcode();
319                                    boolean fin = f.isFin();
320    
321                                    if( curop == Opcode.CLOSING ) {
322                                            int code = CloseFrame.NOCODE;
323                                            String reason = "";
324                                            if( f instanceof CloseFrame ) {
325                                                    CloseFrame cf = (CloseFrame) f;
326                                                    code = cf.getCloseCode();
327                                                    reason = cf.getMessage();
328                                            }
329                                            if( readystate == READYSTATE.CLOSING ) {
330                                                    // complete the close handshake by disconnecting
331                                                    closeConnection( code, reason, true );
332                                            } else {
333                                                    // echo close handshake
334                                                    if( draft.getCloseHandshakeType() == CloseHandshakeType.TWOWAY )
335                                                            close( code, reason, true );
336                                                    else
337                                                            flushAndClose( code, reason, false );
338                                            }
339                                            continue;
340                                    } else if( curop == Opcode.PING ) {
341                                            wsl.onWebsocketPing( this, f );
342                                            continue;
343                                    } else if( curop == Opcode.PONG ) {
344                                            wsl.onWebsocketPong( this, f );
345                                            continue;
346                                    } else if( !fin || curop == Opcode.CONTINUOUS ) {
347                                            if( curop != Opcode.CONTINUOUS ) {
348                                                    if( current_continuous_frame_opcode != null )
349                                                            throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR, "Previous continuous frame sequence not completed." );
350                                                    current_continuous_frame_opcode = curop;
351                                            } else if( fin ) {
352                                                    if( current_continuous_frame_opcode == null )
353                                                            throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR, "Continuous frame sequence was not started." );
354                                                    current_continuous_frame_opcode = null;
355                                            } else if( current_continuous_frame_opcode == null ) {
356                                                    throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR, "Continuous frame sequence was not started." );
357                                            }
358                                            try {
359                                                    wsl.onWebsocketMessageFragment( this, f );
360                                            } catch ( RuntimeException e ) {
361                                                    wsl.onWebsocketError( this, e );
362                                            }
363    
364                                    } else if( current_continuous_frame_opcode != null ) {
365                                            throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR, "Continuous frame sequence not completed." );
366                                    } else if( curop == Opcode.TEXT ) {
367                                            try {
368                                                    wsl.onWebsocketMessage( this, Charsetfunctions.stringUtf8( f.getPayloadData() ) );
369                                            } catch ( RuntimeException e ) {
370                                                    wsl.onWebsocketError( this, e );
371                                            }
372                                    } else if( curop == Opcode.BINARY ) {
373                                            try {
374                                                    wsl.onWebsocketMessage( this, f.getPayloadData() );
375                                            } catch ( RuntimeException e ) {
376                                                    wsl.onWebsocketError( this, e );
377                                            }
378                                    } else {
379                                            throw new InvalidDataException( CloseFrame.PROTOCOL_ERROR, "non control or continious frame expected" );
380                                    }
381                            }
382                    } catch ( InvalidDataException e1 ) {
383                            wsl.onWebsocketError( this, e1 );
384                            close( e1 );
385                            return;
386                    }
387            }

389            private void close( int code, String message, boolean remote ) {
390                    if( readystate != READYSTATE.CLOSING && readystate != READYSTATE.CLOSED ) {
391                            if( readystate == READYSTATE.OPEN ) {
392                                    if( code == CloseFrame.ABNORMAL_CLOSE ) {
393                                            assert ( remote == false );
394                                            readystate = READYSTATE.CLOSING;
395                                            flushAndClose( code, message, false );
396                                            return;
397                                    }
398                                    if( draft.getCloseHandshakeType() != CloseHandshakeType.NONE ) {
399                                            try {
400                                                    if( !remote ) {
401                                                            try {
402                                                                    wsl.onWebsocketCloseInitiated( this, code, message );
403                                                            } catch ( RuntimeException e ) {
404                                                                    wsl.onWebsocketError( this, e );
405                                                            }
406                                                    }
407                                                    sendFrame( new CloseFrameBuilder( code, message ) );
408                                            } catch ( InvalidDataException e ) {
409                                                    wsl.onWebsocketError( this, e );
410                                                    flushAndClose( CloseFrame.ABNORMAL_CLOSE, "generated frame is invalid", false );
411                                            }
412                                    }
413                                    flushAndClose( code, message, remote );
414                            } else if( code == CloseFrame.FLASHPOLICY ) {
415                                    assert ( remote );
416                                    flushAndClose( CloseFrame.FLASHPOLICY, message, true );
417                            } else {
418                                    flushAndClose( CloseFrame.NEVER_CONNECTED, message, false );
419                            }
420                            if( code == CloseFrame.PROTOCOL_ERROR )// this endpoint found a PROTOCOL_ERROR
421                                    flushAndClose( code, message, remote );
422                            readystate = READYSTATE.CLOSING;
423                            tmpHandshakeBytes = null;
424                            return;
425                    }
426            }

433            /**
434             * 
435             * @param remote
436             *            Indicates who "generated" <code>code</code>.<br>
437             *            <code>true</code> means that this endpoint received the <code>code</code> from the other endpoint.<br>
438             *            false means this endpoint decided to send the given code,<br>
439             *            <code>remote</code> may also be true if this endpoint started the closing handshake since the other endpoint may not simply echo the <code>code</code> but close the connection the same time this endpoint does do but with an other <code>code</code>. <br>
440             **/
441    
442            protected synchronized void closeConnection( int code, String message, boolean remote ) {
443                    if( readystate == READYSTATE.CLOSED ) {
444                            return;
445                    }
446    
447                    if( key != null ) {
448                            // key.attach( null ); //see issue #114
449                            key.cancel();
450                    }
451                    if( channel != null ) {
452                            try {
453                                    channel.close();
454                            } catch ( IOException e ) {
455                                    wsl.onWebsocketError( this, e );
456                            }
457                    }
458                    try {
459                            this.wsl.onWebsocketClose( this, code, message, remote );
460                    } catch ( RuntimeException e ) {
461                            wsl.onWebsocketError( this, e );
462                    }
463                    if( draft != null )
464                            draft.reset();
465                    handshakerequest = null;
466    
467                    readystate = READYSTATE.CLOSED;
468                    this.outQueue.clear();
469            }

486            protected synchronized void flushAndClose( int code, String message, boolean remote ) {
487                    if( flushandclosestate ) {
488                            return;
489                    }
490                    closecode = code;
491                    closemessage = message;
492                    closedremotely = remote;
493    
494                    flushandclosestate = true;
495    
496                    wsl.onWriteDemand( this ); // ensures that all outgoing frames are flushed before closing the connection
497                    try {
498                            wsl.onWebsocketClosing( this, code, message, remote );
499                    } catch ( RuntimeException e ) {
500                            wsl.onWebsocketError( this, e );
501                    }
502                    if( draft != null )
503                            draft.reset();
504                    handshakerequest = null;
505            }

507            public void eot() {
508                    if( getReadyState() == READYSTATE.NOT_YET_CONNECTED ) {
509                            closeConnection( CloseFrame.NEVER_CONNECTED, true );
510                    } else if( flushandclosestate ) {
511                            closeConnection( closecode, closemessage, closedremotely );
512                    } else if( draft.getCloseHandshakeType() == CloseHandshakeType.NONE ) {
513                            closeConnection( CloseFrame.NORMAL, true );
514                    } else if( draft.getCloseHandshakeType() == CloseHandshakeType.ONEWAY ) {
515                            if( role == Role.SERVER )
516                                    closeConnection( CloseFrame.ABNORMAL_CLOSE, true );
517                            else
518                                    closeConnection( CloseFrame.NORMAL, true );
519                    } else {
520                            closeConnection( CloseFrame.ABNORMAL_CLOSE, true );
521                    }
522            }

584            private HandshakeState isFlashEdgeCase( ByteBuffer request ) throws IncompleteHandshakeException {
585                    request.mark();
586                    if( request.limit() > Draft.FLASH_POLICY_REQUEST.length ) {
587                            return HandshakeState.NOT_MATCHED;
588                    } else if( request.limit() < Draft.FLASH_POLICY_REQUEST.length ) {
589                            throw new IncompleteHandshakeException( Draft.FLASH_POLICY_REQUEST.length );
590                    } else {
591    
592                            for( int flash_policy_index = 0 ; request.hasRemaining() ; flash_policy_index++ ) {
593                                    if( Draft.FLASH_POLICY_REQUEST[ flash_policy_index ] != request.get() ) {
594                                            request.reset();
595                                            return HandshakeState.NOT_MATCHED;
596                                    }
597                            }
598                            return HandshakeState.MATCHED;
599                    }
600            }

602            public void startHandshake( ClientHandshakeBuilder handshakedata ) throws InvalidHandshakeException {
603                    assert ( readystate != READYSTATE.CONNECTING ) : "shall only be called once";
604    
605                    // Store the Handshake Request we are about to send
606                    this.handshakerequest = draft.postProcessHandshakeRequestAsClient( handshakedata );
607    
608                    // Notify Listener
609                    try {
610                            wsl.onWebsocketHandshakeSentAsClient( this, this.handshakerequest );
611                    } catch ( InvalidDataException e ) {
612                            // Stop if the client code throws an exception
613                            throw new InvalidHandshakeException( "Handshake data rejected by client." );
614                    } catch ( RuntimeException e ) {
615                            wsl.onWebsocketError( this, e );
616                            throw new InvalidHandshakeException( "rejected because of" + e );
617                    }
618    
619                    // Send
620                    write( draft.createHandshake( this.handshakerequest, role ) );
621            }

625                            System.out.println( "write(" + buf.remaining() + "): {" + ( buf.remaining() > 1000 ? "too big to display" : new String( buf.array() ) ) + "}" );

627                    outQueue.add( buf );
628                    /*try {
629                            outQueue.put( buf );

645                    if( DEBUG )
646                            System.out.println( "open using draft: " + draft.getClass().getSimpleName() );
647                    readystate = READYSTATE.OPEN;
