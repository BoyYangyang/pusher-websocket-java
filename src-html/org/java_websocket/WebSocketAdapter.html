001    package org.java_websocket;
002    
003    import org.java_websocket.drafts.Draft;
004    import org.java_websocket.exceptions.InvalidDataException;
005    import org.java_websocket.framing.Framedata;
006    import org.java_websocket.framing.Framedata.Opcode;
007    import org.java_websocket.framing.FramedataImpl1;
008    import org.java_websocket.handshake.ClientHandshake;
009    import org.java_websocket.handshake.HandshakeImpl1Server;
010    import org.java_websocket.handshake.ServerHandshake;
011    import org.java_websocket.handshake.ServerHandshakeBuilder;
012    
013    /**
014     * This class default implements all methods of the WebSocketListener that can be overridden optionally when advances functionalities is needed.<br>
015     **/
016    public abstract class WebSocketAdapter implements WebSocketListener {
017    
018            /**
019             * This default implementation does not do anything. Go ahead and overwrite it.
020             * 
021             * @see org.java_websocket.WebSocketListener#onWebsocketHandshakeReceivedAsServer(WebSocket, Draft, ClientHandshake)
022             */
023            @Override
024            public ServerHandshakeBuilder onWebsocketHandshakeReceivedAsServer( WebSocket conn, Draft draft, ClientHandshake request ) throws InvalidDataException {
025                    return new HandshakeImpl1Server();
026            }
027    
028            @Override
029            public void onWebsocketHandshakeReceivedAsClient( WebSocket conn, ClientHandshake request, ServerHandshake response ) throws InvalidDataException {
030            }
031    
032            /**
033             * This default implementation does not do anything which will cause the connections to always progress.
034             * 
035             * @see org.java_websocket.WebSocketListener#onWebsocketHandshakeSentAsClient(WebSocket, ClientHandshake)
036             */
037            @Override
038            public void onWebsocketHandshakeSentAsClient( WebSocket conn, ClientHandshake request ) throws InvalidDataException {
039            }
040    
041            /**
042             * This default implementation does not do anything. Go ahead and overwrite it
043             * 
044             * @see org.java_websocket.WebSocketListener#onWebsocketMessageFragment(WebSocket, Framedata)
045             */
046            @Override
047            public void onWebsocketMessageFragment( WebSocket conn, Framedata frame ) {
048            }
049    
050            /**
051             * This default implementation will send a pong in response to the received ping.
052             * The pong frame will have the same payload as the ping frame.
053             * 
054             * @see org.java_websocket.WebSocketListener#onWebsocketPing(WebSocket, Framedata)
055             */
056            @Override
057            public void onWebsocketPing( WebSocket conn, Framedata f ) {
058                    FramedataImpl1 resp = new FramedataImpl1( f );
059                    resp.setOptcode( Opcode.PONG );
060                    conn.sendFrame( resp );
061            }
062    
063            /**
064             * This default implementation does not do anything. Go ahead and overwrite it.
065             * 
066             * @see @see org.java_websocket.WebSocketListener#onWebsocketPong(WebSocket, Framedata)
067             */
068            @Override
069            public void onWebsocketPong( WebSocket conn, Framedata f ) {
070            }
071    
072            /**
073             * Gets the XML string that should be returned if a client requests a Flash
074             * security policy.
075             * 
076             * The default implementation allows access from all remote domains, but
077             * only on the port that this WebSocketServer is listening on.
078             * 
079             * This is specifically implemented for gitime's WebSocket client for Flash:
080             * http://github.com/gimite/web-socket-js
081             * 
082             * @return An XML String that comforms to Flash's security policy. You MUST
083             *         not include the null char at the end, it is appended automatically.
084             */
085            @Override
086            public String getFlashPolicy( WebSocket conn ) {
087                    return "<cross-domain-policy><allow-access-from domain=\"*\" to-ports=\"" + conn.getLocalSocketAddress().getPort() + "\" /></cross-domain-policy>\0";
088            }
089    
090    }
