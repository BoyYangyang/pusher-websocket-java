001    package org.java_websocket;
002    
003    import java.net.InetSocketAddress;
004    import java.nio.ByteBuffer;
005    import java.nio.channels.NotYetConnectedException;
006    
007    import org.java_websocket.drafts.Draft;
008    import org.java_websocket.framing.Framedata;
009    
010    public interface WebSocket {
011            public enum Role {
012                    CLIENT, SERVER
013            }
014    
015            public enum READYSTATE {
016                    NOT_YET_CONNECTED, CONNECTING, OPEN, CLOSING, CLOSED;
017            }
018    
019            /**
020             * The default port of WebSockets, as defined in the spec. If the nullary
021             * constructor is used, DEFAULT_PORT will be the port the WebSocketServer
022             * is binded to. Note that ports under 1024 usually require root permissions.
023             */
024            public static final int DEFAULT_PORT = 80;
025    
026            public static final int DEFAULT_WSS_PORT = 443;
027    
028            /**
029             * sends the closing handshake.
030             * may be send in response to an other handshake.
031             */
032            public void close( int code, String message );
033    
034            public void close( int code );
035    
036            /** Convenience function which behaves like close(CloseFrame.NORMAL) */
037            public void close();
038    
039            /**
040             * This will close the connection immediately without a proper close handshake.
041             * The code and the message therefore won't be transfered over the wire also they will be forwarded to onClose/onWebsocketClose.
042             **/
043            public abstract void closeConnection( int code, String message );
044    
045            /**
046             * Send Text data to the other end.
047             * 
048             * @throws IllegalArgumentException
049             * @throws NotYetConnectedException
050             */
051            public abstract void send( String text ) throws NotYetConnectedException;
052    
053            /**
054             * Send Binary data (plain bytes) to the other end.
055             * 
056             * @throws IllegalArgumentException
057             * @throws NotYetConnectedException
058             */
059            public abstract void send( ByteBuffer bytes ) throws IllegalArgumentException , NotYetConnectedException;
060    
061            public abstract void send( byte[] bytes ) throws IllegalArgumentException , NotYetConnectedException;
062    
063            public abstract void sendFrame( Framedata framedata );
064    
065            public abstract boolean hasBufferedData();
066    
067            /**
068             * @returns never returns null
069             */
070            public abstract InetSocketAddress getRemoteSocketAddress();
071    
072            /**
073             * @returns never returns null
074             */
075            public abstract InetSocketAddress getLocalSocketAddress();
076    
077            public abstract boolean isConnecting();
078    
079            public abstract boolean isOpen();
080    
081            public abstract boolean isClosing();
082    
083            /**
084             * Returns true when no further frames may be submitted<br>
085             * This happens before the socket connection is closed.
086             */
087            public abstract boolean isFlushAndClose();
088    
089            /** Returns whether the close handshake has been completed and the socket is closed. */
090            public abstract boolean isClosed();
091    
092            public abstract Draft getDraft();
093    
094            /**
095             * Retrieve the WebSocket 'readyState'.
096             * This represents the state of the connection.
097             * It returns a numerical value, as per W3C WebSockets specs.
098             * 
099             * @return Returns '0 = CONNECTING', '1 = OPEN', '2 = CLOSING' or '3 = CLOSED'
100             */
101            public abstract READYSTATE getReadyState();
102    }
