001    package org.java_websocket.framing;
002    
003    import org.java_websocket.exceptions.InvalidDataException;
004    import org.java_websocket.exceptions.InvalidFrameException;
005    
006    public interface CloseFrame extends Framedata {
007            /**
008             * indicates a normal closure, meaning whatever purpose the
009             * connection was established for has been fulfilled.
010             */
011            public static final int NORMAL = 1000;
012            /**
013             * 1001 indicates that an endpoint is "going away", such as a server
014             * going down, or a browser having navigated away from a page.
015             */
016            public static final int GOING_AWAY = 1001;
017            /**
018             * 1002 indicates that an endpoint is terminating the connection due
019             * to a protocol error.
020             */
021            public static final int PROTOCOL_ERROR = 1002;
022            /**
023             * 1003 indicates that an endpoint is terminating the connection
024             * because it has received a type of data it cannot accept (e.g. an
025             * endpoint that understands only text data MAY send this if it
026             * receives a binary message).
027             */
028            public static final int REFUSE = 1003;
029            /*1004: Reserved. The specific meaning might be defined in the future.*/
030            /**
031             * 1005 is a reserved value and MUST NOT be set as a status code in a
032             * Close control frame by an endpoint. It is designated for use in
033             * applications expecting a status code to indicate that no status
034             * code was actually present.
035             */
036            public static final int NOCODE = 1005;
037            /**
038             * 1006 is a reserved value and MUST NOT be set as a status code in a
039             * Close control frame by an endpoint. It is designated for use in
040             * applications expecting a status code to indicate that the
041             * connection was closed abnormally, e.g. without sending or
042             * receiving a Close control frame.
043             */
044            public static final int ABNORMAL_CLOSE = 1006;
045            /**
046             * 1007 indicates that an endpoint is terminating the connection
047             * because it has received data within a message that was not
048             * consistent with the type of the message (e.g., non-UTF-8 [RFC3629]
049             * data within a text message).
050             */
051            public static final int NO_UTF8 = 1007;
052            /**
053             * 1008 indicates that an endpoint is terminating the connection
054             * because it has received a message that violates its policy. This
055             * is a generic status code that can be returned when there is no
056             * other more suitable status code (e.g. 1003 or 1009), or if there
057             * is a need to hide specific details about the policy.
058             */
059            public static final int POLICY_VALIDATION = 1008;
060            /**
061             * 1009 indicates that an endpoint is terminating the connection
062             * because it has received a message which is too big for it to
063             * process.
064             */
065            public static final int TOOBIG = 1009;
066            /**
067             * 1010 indicates that an endpoint (client) is terminating the
068             * connection because it has expected the server to negotiate one or
069             * more extension, but the server didn't return them in the response
070             * message of the WebSocket handshake. The list of extensions which
071             * are needed SHOULD appear in the /reason/ part of the Close frame.
072             * Note that this status code is not used by the server, because it
073             * can fail the WebSocket handshake instead.
074             */
075            public static final int EXTENSION = 1010;
076            /**
077             * 1011 indicates that a server is terminating the connection because
078             * it encountered an unexpected condition that prevented it from
079             * fulfilling the request.
080             **/
081            public static final int UNEXPECTED_CONDITION = 1011;
082            /**
083             * 1015 is a reserved value and MUST NOT be set as a status code in a
084             * Close control frame by an endpoint. It is designated for use in
085             * applications expecting a status code to indicate that the
086             * connection was closed due to a failure to perform a TLS handshake
087             * (e.g., the server certificate can't be verified).
088             **/
089            public static final int TLS_ERROR = 1015;
090    
091            /** The connection had never been established */
092            public static final int NEVER_CONNECTED = -1;
093            public static final int BUGGYCLOSE = -2;
094            public static final int FLASHPOLICY = -3;
095    
096            public int getCloseCode() throws InvalidFrameException;
097            public String getMessage() throws InvalidDataException;
098    }
