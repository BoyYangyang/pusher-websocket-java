001    package org.java_websocket;
002    
003    import java.net.InetSocketAddress;
004    import java.nio.ByteBuffer;
005    
006    import org.java_websocket.drafts.Draft;
007    import org.java_websocket.exceptions.InvalidDataException;
008    import org.java_websocket.framing.Framedata;
009    import org.java_websocket.handshake.ClientHandshake;
010    import org.java_websocket.handshake.Handshakedata;
011    import org.java_websocket.handshake.ServerHandshake;
012    import org.java_websocket.handshake.ServerHandshakeBuilder;
013    
014    /**
015     * Implemented by <tt>WebSocketClient</tt> and <tt>WebSocketServer</tt>.
016     * The methods within are called by <tt>WebSocket</tt>.
017     * Almost every method takes a first parameter conn which represents the source of the respective event.
018     */
019    public interface WebSocketListener {
020    
021            /**
022             * Called on the server side when the socket connection is first established, and the WebSocket
023             * handshake has been received. This method allows to deny connections based on the received handshake.<br>
024             * By default this method only requires protocol compliance.
025             * 
026             * @param conn
027             *            The WebSocket related to this event
028             * @param draft
029             *            The protocol draft the client uses to connect
030             * @param request
031             *            The opening http message send by the client. Can be used to access additional fields like cookies.
032             * @return Returns an incomplete handshake containing all optional fields
033             * @throws InvalidDataException
034             *             Throwing this exception will cause this handshake to be rejected
035             */
036            public ServerHandshakeBuilder onWebsocketHandshakeReceivedAsServer( WebSocket conn, Draft draft, ClientHandshake request ) throws InvalidDataException;
037    
038            /**
039             * Called on the client side when the socket connection is first established, and the WebSocketImpl
040             * handshake response has been received.
041             * 
042             * @param conn
043             *            The WebSocket related to this event
044             * @param request
045             *            The handshake initially send out to the server by this websocket.
046             * @param response
047             *            The handshake the server sent in response to the request.
048             * @throws InvalidDataException
049             *             Allows the client to reject the connection with the server in respect of its handshake response.
050             */
051            public void onWebsocketHandshakeReceivedAsClient( WebSocket conn, ClientHandshake request, ServerHandshake response ) throws InvalidDataException;
052    
053            /**
054             * Called on the client side when the socket connection is first established, and the WebSocketImpl
055             * handshake has just been sent.
056             * 
057             * @param conn
058             *            The WebSocket related to this event
059             * @param request
060             *            The handshake sent to the server by this websocket
061             * @throws InvalidDataException
062             *             Allows the client to stop the connection from progressing
063             */
064            public void onWebsocketHandshakeSentAsClient( WebSocket conn, ClientHandshake request ) throws InvalidDataException;
065    
066            /**
067             * Called when an entire text frame has been received. Do whatever you want
068             * here...
069             * 
070             * @param conn
071             *            The <tt>WebSocket</tt> instance this event is occurring on.
072             * @param message
073             *            The UTF-8 decoded message that was received.
074             */
075            public void onWebsocketMessage( WebSocket conn, String message );
076    
077            /**
078             * Called when an entire binary frame has been received. Do whatever you want
079             * here...
080             * 
081             * @param conn
082             *            The <tt>WebSocket</tt> instance this event is occurring on.
083             * @param blob
084             *            The binary message that was received.
085             */
086            public void onWebsocketMessage( WebSocket conn, ByteBuffer blob );
087    
088            public void onWebsocketMessageFragment( WebSocket conn, Framedata frame );
089    
090            /**
091             * Called after <var>onHandshakeReceived</var> returns <var>true</var>.
092             * Indicates that a complete WebSocket connection has been established,
093             * and we are ready to send/receive data.
094             * 
095             * @param conn
096             *            The <tt>WebSocket</tt> instance this event is occuring on.
097             */
098            public void onWebsocketOpen( WebSocket conn, Handshakedata d );
099    
100            /**
101             * Called after <tt>WebSocket#close</tt> is explicity called, or when the
102             * other end of the WebSocket connection is closed.
103             * 
104             * @param conn
105             *            The <tt>WebSocket</tt> instance this event is occuring on.
106             */
107            public void onWebsocketClose( WebSocket ws, int code, String reason, boolean remote );
108    
109            /** called as soon as no further frames are accepted */
110            public void onWebsocketClosing( WebSocket ws, int code, String reason, boolean remote );
111    
112            /** send when this peer sends a close handshake */
113            public void onWebsocketCloseInitiated( WebSocket ws, int code, String reason );
114    
115            /**
116             * Called if an exception worth noting occurred.
117             * If an error causes the connection to fail onClose will be called additionally afterwards.
118             * 
119             * @param ex
120             *            The exception that occurred. <br>
121             *            Might be null if the exception is not related to any specific connection. For example if the server port could not be bound.
122             */
123            public void onWebsocketError( WebSocket conn, Exception ex );
124    
125            /**
126             * Called a ping frame has been received.
127             * This method must send a corresponding pong by itself.
128             * 
129             * @param f
130             *            The ping frame. Control frames may contain payload.
131             */
132            public void onWebsocketPing( WebSocket conn, Framedata f );
133    
134            /**
135             * Called when a pong frame is received.
136             **/
137            public void onWebsocketPong( WebSocket conn, Framedata f );
138    
139            /**
140             * Gets the XML string that should be returned if a client requests a Flash
141             * security policy.
142             */
143            public String getFlashPolicy( WebSocket conn );
144    
145            /** This method is used to inform the selector thread that there is data queued to be written to the socket. */
146            public void onWriteDemand( WebSocket conn );
147    
148            public InetSocketAddress getLocalSocketAddress( WebSocket conn );
149            public InetSocketAddress getRemoteSocketAddress( WebSocket conn );
150    }
