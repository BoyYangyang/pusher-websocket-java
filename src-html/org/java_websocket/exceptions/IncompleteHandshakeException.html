001    package org.java_websocket.exceptions;
002    
003    public class IncompleteHandshakeException extends RuntimeException {
004    
005            private static final long serialVersionUID = 7906596804233893092L;
006            private int newsize;
007    
008            public IncompleteHandshakeException( int newsize ) {
009                    this.newsize = newsize;
010            }
011    
012            public IncompleteHandshakeException() {
013                    this.newsize = 0;
014            }
015    
016            public int getPreferedSize() {
017                    return newsize;
018            }
019    
020    }
