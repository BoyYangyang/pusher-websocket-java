030    /**
031     * Implements the relevant portions of the SocketChannel interface with the SSLEngine wrapper.
032     */
033    public class SSLSocketChannel2 implements ByteChannel, WrappedByteChannel {
034            protected static ByteBuffer emptybuffer = ByteBuffer.allocate( 0 );
035    
036            protected ExecutorService exec;
037    
038            protected List<Future<?>> tasks;
039    
040            /** raw payload incomming */
041            protected ByteBuffer inData;
042            /** encrypted data outgoing */
043            protected ByteBuffer outCrypt;
044            /** encrypted data incoming */
045            protected ByteBuffer inCrypt;
046    
047            /** the underlying channel */
048            protected SocketChannel socketChannel;
049            /** used to set interestOP SelectionKey.OP_WRITE for the underlying channel */
050            protected SelectionKey selectionKey;
051    
052    
053            protected SSLEngineResult engineResult;
054            protected SSLEngine sslEngine;
055    
056    
057            private Status engineStatus = Status.BUFFER_UNDERFLOW;
058    
059            public SSLSocketChannel2( SocketChannel channel , SSLEngine sslEngine , ExecutorService exec , SelectionKey key ) throws IOException {
060                    if( channel == null || sslEngine == null || exec == null )
061                            throw new IllegalArgumentException( "parameter must not be null" );
062    
063                    this.socketChannel = channel;
064                    this.sslEngine = sslEngine;
065                    this.exec = exec;
066    
067                    tasks = new ArrayList<Future<?>>( 3 );
068                    if( key != null ) {
069                            key.interestOps( key.interestOps() | SelectionKey.OP_WRITE );
070                            this.selectionKey = key;
071                    }
072                    createBuffers( sslEngine.getSession() );
073                    // kick off handshake
074                    socketChannel.write( wrap( emptybuffer ) );// initializes res
075                    processHandshake();
076            }
077    
078            private void consumeFutureUninterruptible( Future<?> f ) {
079                    try {
080                            boolean interrupted = false;
081                            while ( true ) {
082                                    try {
083                                            f.get();
084                                            break;
085                                    } catch ( InterruptedException e ) {
086                                            interrupted = true;
087                                    }
088                            }
089                            if( interrupted )
090                                    Thread.currentThread().interrupt();
091                    } catch ( ExecutionException e ) {
092                            throw new RuntimeException( e );
093                    }
094            }
095    
096            private synchronized void processHandshake() throws IOException {
097                    if( engineResult.getHandshakeStatus() == HandshakeStatus.NOT_HANDSHAKING )
098                            return; // since this may be called either from a reading or a writing thread and because this method is synchronized it is necessary to double check if we are still handshaking.
099                    if( !tasks.isEmpty() ) {
100                            Iterator<Future<?>> it = tasks.iterator();
101                            while ( it.hasNext() ) {
102                                    Future<?> f = it.next();
103                                    if( f.isDone() ) {
104                                            it.remove();
105                                    } else {
106                                            if( isBlocking() )
107                                                    consumeFutureUninterruptible( f );
108                                            return;
109                                    }
110                            }
111                    }
112    
113                    if( engineResult.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.NEED_UNWRAP ) {
114                            if( !isBlocking() || engineStatus == Status.BUFFER_UNDERFLOW ) {
115                                    inCrypt.compact();
116                                    int read = socketChannel.read( inCrypt );
117                                    if( read == -1 ) {
118                                            throw new IOException( "connection closed unexpectedly by peer" );
119                                    }
120                                    inCrypt.flip();
121                            }
122                            inData.compact();
123                            unwrap();
124                            if( engineResult.getHandshakeStatus() == HandshakeStatus.FINISHED ) {
125                                    createBuffers( sslEngine.getSession() );
126                                    return;
127                            }
128                    }
129                    consumeDelegatedTasks();
130                    assert ( engineResult.getHandshakeStatus() != HandshakeStatus.NOT_HANDSHAKING );
131                    if( tasks.isEmpty() || engineResult.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.NEED_WRAP ) {
132                            socketChannel.write( wrap( emptybuffer ) );
133                            if( engineResult.getHandshakeStatus() == HandshakeStatus.FINISHED ) {
134                                    createBuffers( sslEngine.getSession() );
135                            }
136                    }
137            }
138    
139            private synchronized ByteBuffer wrap( ByteBuffer b ) throws SSLException {
140                    outCrypt.compact();
141                    engineResult = sslEngine.wrap( b, outCrypt );
142                    outCrypt.flip();
143                    return outCrypt;
144            }
145    
146            private synchronized ByteBuffer unwrap() throws SSLException {
147                    int rem;
148                    do {
149                            rem = inData.remaining();
150                            engineResult = sslEngine.unwrap( inCrypt, inData );
151                            engineStatus = engineResult.getStatus();
152                    } while ( engineStatus == SSLEngineResult.Status.OK && ( rem != inData.remaining() || engineResult.getHandshakeStatus() == HandshakeStatus.NEED_UNWRAP ) );
153    
154                    inData.flip();
155                    return inData;
156            }
157    
158            protected void consumeDelegatedTasks() {
159                    Runnable task;
160                    while ( ( task = sslEngine.getDelegatedTask() ) != null ) {
161                            tasks.add( exec.submit( task ) );
162                            // task.run();
163                    }
164            }
165    
166            protected void createBuffers( SSLSession session ) {
167                    int appBufferMax = session.getApplicationBufferSize();
168                    int netBufferMax = session.getPacketBufferSize();
169    
170                    if( inData == null ) {
171                            inData = ByteBuffer.allocate( appBufferMax );
172                            outCrypt = ByteBuffer.allocate( netBufferMax );
173                            inCrypt = ByteBuffer.allocate( netBufferMax );
174                    } else {
175                            if( inData.capacity() != appBufferMax )
176                                    inData = ByteBuffer.allocate( appBufferMax );
177                            if( outCrypt.capacity() != netBufferMax )
178                                    outCrypt = ByteBuffer.allocate( netBufferMax );
179                            if( inCrypt.capacity() != netBufferMax )
180                                    inCrypt = ByteBuffer.allocate( netBufferMax );
181                    }
182                    inData.rewind();
183                    inData.flip();
184                    inCrypt.rewind();
185                    inCrypt.flip();
186                    outCrypt.rewind();
187                    outCrypt.flip();
188            }
189    
190            public int write( ByteBuffer src ) throws IOException {
191                    if( !isHandShakeComplete() ) {
192                            processHandshake();
193                            return 0;
194                    }
195                    int num = socketChannel.write( wrap( src ) );
196                    return num;
197    
198            }
199    
200            /**
201             * Blocks when in blocking mode until at least one byte has been decoded.<br>
202             * When not in blocking mode 0 may be returned.
203             **/
204            public int read( ByteBuffer dst ) throws IOException {
205                    if( !dst.hasRemaining() )
206                            return 0;
207                    if( !isHandShakeComplete() ) {
208                            if( isBlocking() ) {
209                                    while ( !isHandShakeComplete() ) {
210                                            processHandshake();
211                                    }
212                            } else {
213                                    processHandshake();
214                                    if( !isHandShakeComplete() ) {
215                                            return 0;
216                                    }
217                            }
218                    }
219    
220                    int purged = readRemaining( dst );
221                    if( purged != 0 )
222                            return purged;
223    
224                    assert ( inData.position() == 0 );
225                    inData.clear();
226    
227                    if( !inCrypt.hasRemaining() )
228                            inCrypt.clear();
229                    else
230                            inCrypt.compact();
231    
232                    if( ( isBlocking() && inCrypt.position() == 0 ) || engineStatus == Status.BUFFER_UNDERFLOW )
233                            if( socketChannel.read( inCrypt ) == -1 ) {
234                                    return -1;
235                            }
236                    inCrypt.flip();
237                    unwrap();
238    
239                    int transfered = transfereTo( inData, dst );
240                    if( transfered == 0 && isBlocking() ) {
241                            return read( dst ); // "transfered" may be 0 when not enough bytes were received or during rehandshaking
242                    }
243                    return transfered;
244            }
245    
246            private int readRemaining( ByteBuffer dst ) throws SSLException {
247                    if( inData.hasRemaining() ) {
248                            return transfereTo( inData, dst );
249                    }
250                    if( !inData.hasRemaining() )
251                            inData.clear();
252                    // test if some bytes left from last read (e.g. BUFFER_UNDERFLOW)
253                    if( inCrypt.hasRemaining() ) {
254                            unwrap();
255                            int amount = transfereTo( inData, dst );
256                            if( amount > 0 )
257                                    return amount;
258                    }
259                    return 0;
260            }

262            public boolean isConnected() {
263                    return socketChannel.isConnected();
264            }
265    
266            public void close() throws IOException {
267                    sslEngine.closeOutbound();
268                    sslEngine.getSession().invalidate();
269                    if( socketChannel.isOpen() )
270                            socketChannel.write( wrap( emptybuffer ) );// FIXME what if not all bytes can be written
271                    socketChannel.close();
272            }
273    
274            private boolean isHandShakeComplete() {
275                    HandshakeStatus status = engineResult.getHandshakeStatus();
276                    return status == SSLEngineResult.HandshakeStatus.FINISHED || status == SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING;
277            }
278    
279            public SelectableChannel configureBlocking( boolean b ) throws IOException {
280                    return socketChannel.configureBlocking( b );
281            }
282    
283            public boolean connect( SocketAddress remote ) throws IOException {
284                    return socketChannel.connect( remote );
285            }
286    
287            public boolean finishConnect() throws IOException {
288                    return socketChannel.finishConnect();
289            }
290    
291            public Socket socket() {
292                    return socketChannel.socket();
293            }
294    
295            public boolean isInboundDone() {
296                    return sslEngine.isInboundDone();
297            }
298    
299            @Override
300            public boolean isOpen() {
301                    return socketChannel.isOpen();
302            }
303    
304            @Override
305            public boolean isNeedWrite() {
306                    return outCrypt.hasRemaining() || !isHandShakeComplete();
307            }
308    
309            @Override
310            public void writeMore() throws IOException {
311                    write( outCrypt );
312            }
313    
314            @Override
315            public boolean isNeedRead() {
316                    return inData.hasRemaining() || ( inCrypt.hasRemaining() && engineResult.getStatus() != Status.BUFFER_UNDERFLOW );
317            }
318    
319            @Override
320            public int readMore( ByteBuffer dst ) throws SSLException {
321                    return readRemaining( dst );
322            }
323    
324            private int transfereTo( ByteBuffer from, ByteBuffer to ) {
325                    int fremain = from.remaining();
326                    int toremain = to.remaining();
327                    if( fremain > toremain ) {
328                            // FIXME there should be a more efficient transfer method
329                            int limit = Math.min( fremain, toremain );
330                            for( int i = 0 ; i < limit ; i++ ) {
331                                    to.put( from.get() );
332                            }
333                            return limit;
334                    } else {
335                            to.put( from );
336                            return fremain;
337                    }
338    
339            }
340    
341            @Override
342            public boolean isBlocking() {
343                    return socketChannel.isBlocking();
344            }
