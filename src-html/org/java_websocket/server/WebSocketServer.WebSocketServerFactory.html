127            /**
128             * Creates a WebSocketServer that will attempt to bind/listen on the given <var>address</var>,
129             * and comply with <tt>Draft</tt> version <var>draft</var>.
130             * 
131             * @param address
132             *            The address (host:port) this server should listen on.
133             * @param decodercount
134             *            The number of {@link WebSocketWorker}s that will be used to process the incoming network data. By default this will be <code>Runtime.getRuntime().availableProcessors()</code>
135             * @param drafts
136             *            The versions of the WebSocket protocol that this server
137             *            instance should comply to. Clients that use an other protocol version will be rejected.
138             * 
139             * @param connectionscontainer
140             *            Allows to specify a collection that will be used to store the websockets in. <br>
141             *            If you plan to often iterate through the currently connected websockets you may want to use a collection that does not require synchronization like a {@link CopyOnWriteArraySet}. In that case make sure that you overload {@link #removeConnection(WebSocket)} and {@link #addConnection(WebSocket)}.<br>
142             *            By default a {@link HashSet} will be used.
143             * 
144             * @see #removeConnection(WebSocket) for more control over syncronized operation
145             * @see <a href="https://github.com/TooTallNate/Java-WebSocket/wiki/Drafts" > more about drafts
146             */
147            public WebSocketServer( InetSocketAddress address , int decodercount , List<Draft> drafts , Collection<WebSocket> connectionscontainer ) {
148                    if( address == null || decodercount < 1 || connectionscontainer == null ) {
149                            throw new IllegalArgumentException( "address and connectionscontainer must not be null and you need at least 1 decoder" );
150                    }
151    
152                    if( drafts == null )
153                            this.drafts = Collections.emptyList();
154                    else
155                            this.drafts = drafts;
156    
157                    this.address = address;
158                    this.connections = connectionscontainer;
159    
160                    iqueue = new LinkedList<WebSocketImpl>();
161    
162                    decoders = new ArrayList<WebSocketWorker>( decodercount );
163                    buffers = new LinkedBlockingQueue<ByteBuffer>();
164                    for( int i = 0 ; i < decodercount ; i++ ) {
165                            WebSocketWorker ex = new WebSocketWorker();
166                            decoders.add( ex );
167                            ex.start();
168                    }
169            }

172             * Starts the server selectorthread that binds to the currently set port number and

174             * May only be called once.

176             * Alternatively you can call {@link WebSocketServer#run()} directly.

178             * @throws IllegalStateException
179             */

181                    if( selectorthread != null )
182                            throw new IllegalStateException( getClass().getName() + " can only be started once." );
183                    new Thread( this ).start();;
184            }

186            /**
187             * Closes all connected clients sockets, then closes the underlying
188             * ServerSocketChannel, effectively killing the server socket selectorthread,
189             * freeing the port the server was bound to and stops all internal workerthreads.
190             * 
191             * If this method is called before the server is started it will never start.
192             * 
193             * @param timeout
194             *            Specifies how many milliseconds shall pass between initiating the close handshakes with the connected clients and closing the servers socket channel.
195             * 
196             * @throws IOException
197             *             When {@link ServerSocketChannel}.close throws an IOException
198             * @throws InterruptedException
199             */
200            public void stop( int timeout ) throws IOException , InterruptedException {
201                    if( !isclosed.compareAndSet( false, true ) ) {
202                            return;
203                    }
204    
205                    synchronized ( connections ) {
206                            for( WebSocket ws : connections ) {
207                                    ws.close( CloseFrame.GOING_AWAY );
208                            }
209                    }
210                    synchronized ( this ) {
211                            if( selectorthread != null ) {
212                                    if( Thread.currentThread() != selectorthread ) {
213    
214                                    }
215                                    if( selectorthread != Thread.currentThread() ) {
216                                            selectorthread.interrupt();
217                                            selectorthread.join();
218                                    }
219                            }
220                            if( decoders != null ) {
221                                    for( WebSocketWorker w : decoders ) {
222                                            w.interrupt();
223                                    }
224                            }
225                            if( server != null ) {
226                                    server.close();
227                            }
228                    }
229            }

231            public void stop() throws IOException , InterruptedException {
232                    stop( 0 );
233            }

267            // Runnable IMPLEMENTATION /////////////////////////////////////////////////
268            public void run() {
269                    synchronized ( this ) {
270                            if( selectorthread != null )
271                                    throw new IllegalStateException( getClass().getName() + " can only be started once." );
272                            selectorthread = Thread.currentThread();
273                            if( isclosed.get() ) {
274                                    return;
275                            }
276                    }
277                    selectorthread.setName( "WebsocketSelector" + selectorthread.getId() );
278                    try {
279                            server = ServerSocketChannel.open();
280                            server.configureBlocking( false );
281                            ServerSocket socket = server.socket();
282                            socket.setReceiveBufferSize( WebSocketImpl.RCVBUF );
283                            socket.bind( address );
284                            selector = Selector.open();
285                            server.register( selector, server.validOps() );
286                    } catch ( IOException ex ) {
287                            handleFatal( null, ex );
288                            return;
289                    }
290                    try {
291                            while ( !selectorthread.isInterrupted() ) {
292                                    SelectionKey key = null;
293                                    WebSocketImpl conn = null;
294                                    try {
295                                            selector.select();
296                                            Set<SelectionKey> keys = selector.selectedKeys();
297                                            Iterator<SelectionKey> i = keys.iterator();
298    
299                                            while ( i.hasNext() ) {
300                                                    key = i.next();
301    
302                                                    if( !key.isValid() ) {
303                                                            // Object o = key.attachment();
304                                                            continue;
305                                                    }
306    
307                                                    if( key.isAcceptable() ) {
308                                                            if( !onConnect( key ) ) {
309                                                                    key.cancel();
310                                                                    continue;
311                                                            }
312    
313                                                            SocketChannel channel = server.accept();
314                                                            channel.configureBlocking( false );
315                                                            WebSocketImpl w = wsf.createWebSocket( this, drafts, channel.socket() );
316                                                            w.key = channel.register( selector, SelectionKey.OP_READ, w );
317                                                            w.channel = wsf.wrapChannel( channel, w.key );
318                                                            i.remove();
319                                                            allocateBuffers( w );
320                                                            continue;
321                                                    }
322    
323                                                    if( key.isReadable() ) {
324                                                            conn = (WebSocketImpl) key.attachment();
325                                                            ByteBuffer buf = takeBuffer();
326                                                            try {
327                                                                    if( SocketChannelIOHelper.read( buf, conn, (ByteChannel) conn.channel ) ) {
328                                                                            conn.inQueue.put( buf );
329                                                                            queue( conn );
330                                                                            i.remove();
331                                                                            if( conn.channel instanceof WrappedByteChannel ) {
332                                                                                    if( ( (WrappedByteChannel) conn.channel ).isNeedRead() ) {
333                                                                                            iqueue.add( conn );
334                                                                                    }
335                                                                            }
336                                                                    } else {
337                                                                            pushBuffer( buf );
338                                                                    }
339                                                            } catch ( IOException e ) {
340                                                                    pushBuffer( buf );
341                                                                    throw e;
342                                                            } catch ( RuntimeException e ) {
343                                                                    pushBuffer( buf );
344                                                                    throw e;
345                                                            }
346                                                    }
347                                                    if( key.isWritable() ) {
348                                                            conn = (WebSocketImpl) key.attachment();
349                                                            if( SocketChannelIOHelper.batch( conn, (ByteChannel) conn.channel ) ) {
350                                                                    if( key.isValid() )
351                                                                            key.interestOps( SelectionKey.OP_READ );
352                                                            }
353                                                    }
354                                            }
355                                            while ( !iqueue.isEmpty() ) {
356                                                    conn = iqueue.remove( 0 );
357                                                    WrappedByteChannel c = ( (WrappedByteChannel) conn.channel );
358                                                    ByteBuffer buf = takeBuffer();
359                                                    try {
360                                                            if( SocketChannelIOHelper.readMore( buf, conn, c ) )
361                                                                    iqueue.add( conn );
362                                                            conn.inQueue.put( buf );
363                                                            queue( conn );
364                                                    } finally {
365                                                            pushBuffer( buf );
366                                                    }
367    
368                                            }
369                                    } catch ( CancelledKeyException e ) {
370                                            // an other thread may cancel the key
371                                    } catch ( IOException ex ) {
372                                            if( key != null )
373                                                    key.cancel();
374                                            handleIOException( key, conn, ex );
375                                    } catch ( InterruptedException e ) {
376                                            return;// FIXME controlled shutdown
377                                    }
378                            }
379    
380                    } catch ( RuntimeException e ) {
381                            // should hopefully never occur
382                            handleFatal( null, e );
383                    }
384            }
385            protected void allocateBuffers( WebSocket c ) throws InterruptedException {
386                    if( queuesize.get() >= 2 * decoders.size() + 1 ) {
387                            return;
388                    }
389                    queuesize.incrementAndGet();
390                    buffers.put( createBuffer() );
391            }
392    
393            protected void releaseBuffers( WebSocket c ) throws InterruptedException {
394                    // queuesize.decrementAndGet();
395                    // takeBuffer();
396            }

398            public ByteBuffer createBuffer() {
399                    return ByteBuffer.allocate( WebSocketImpl.RCVBUF );
400            }

402            private void queue( WebSocketImpl ws ) throws InterruptedException {
403                    if( ws.workerThread == null ) {
404                            ws.workerThread = decoders.get( queueinvokes % decoders.size() );
405                            queueinvokes++;
406                    }
407                    ws.workerThread.put( ws );
408            }
409    
410            private ByteBuffer takeBuffer() throws InterruptedException {
411                    return buffers.take();
412            }
413    
414            private void pushBuffer( ByteBuffer buf ) throws InterruptedException {
415                    if( buffers.size() > queuesize.intValue() )
416                            return;
417                    buffers.put( buf );
418            }
419    
420            private void handleIOException( SelectionKey key, WebSocket conn, IOException ex ) {
421                    onWebsocketError( conn, ex );// conn may be null here
422                    if( conn != null ) {
423                            conn.closeConnection( CloseFrame.ABNORMAL_CLOSE, ex.getMessage() );
424                    } else if( key != null ) {
425                            SelectableChannel channel = key.channel();
426                            if( channel != null && channel.isOpen() ) { // this could be the case if the IOException ex is a SSLException
427                                    try {
428                                            channel.close();
429                                    } catch ( IOException e ) {
430                                            // there is nothing that must be done here
431                                    }
432                                    if( WebSocketImpl.DEBUG )
433                                            System.out.println( "Connection closed because of" + ex );
434                            }
435                    }
436            }
437    
438            private void handleFatal( WebSocket conn, Exception e ) {
439                    onError( conn, e );
440                    try {
441                            stop();
442                    } catch ( IOException e1 ) {
443                            onError( null, e1 );
444                    } catch ( InterruptedException e1 ) {
445                            Thread.currentThread().interrupt();
446                            onError( null, e1 );
447                    }
448            }

486                    selector.wakeup();
487                    try {
488                            if( removeConnection( conn ) ) {
489                                    onClose( conn, code, reason, remote );
490                            }
491                    } finally {
492                            try {
493                                    releaseBuffers( conn );
494                            } catch ( InterruptedException e ) {
495                                    Thread.currentThread().interrupt();
496                            }
497                    }

530            public final void onWriteDemand( WebSocket w ) {
531                    WebSocketImpl conn = (WebSocketImpl) w;
532                    try {
533                            conn.key.interestOps( SelectionKey.OP_READ | SelectionKey.OP_WRITE );
534                    } catch ( CancelledKeyException e ) {
535                            // the thread which cancels key is responsible for possible cleanup
536                            conn.outQueue.clear();
537                    }
538                    selector.wakeup();
539            }

628            public class WebSocketWorker extends Thread {
629    
630                    private BlockingQueue<WebSocketImpl> iqueue;
631    
632                    public WebSocketWorker() {
633                            iqueue = new LinkedBlockingQueue<WebSocketImpl>();
634                            setName( "WebSocketWorker-" + getId() );
635                            setUncaughtExceptionHandler( new UncaughtExceptionHandler() {
636                                    @Override
637                                    public void uncaughtException( Thread t, Throwable e ) {
638                                            getDefaultUncaughtExceptionHandler().uncaughtException( t, e );
639                                    }
640                            } );
641                    }
642    
643                    public void put( WebSocketImpl ws ) throws InterruptedException {
644                            iqueue.put( ws );
645                    }
646    
647                    @Override
648                    public void run() {
649                            WebSocketImpl ws = null;
650                            try {
651                                    while ( true ) {
652                                            ByteBuffer buf = null;
653                                            ws = iqueue.take();
654                                            buf = ws.inQueue.poll();
655                                            assert ( buf != null );
656                                            try {
657                                                    ws.decode( buf );
658                                            } finally {
659                                                    pushBuffer( buf );
660                                            }
661                                    }
662                            } catch ( InterruptedException e ) {
663                            } catch ( RuntimeException e ) {
664                                    handleFatal( ws, e );
665                            }
666                    }
667            }
